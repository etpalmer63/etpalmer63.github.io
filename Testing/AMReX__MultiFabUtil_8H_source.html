295template <typename FAB>
296void average_down_nodal (const FabArray<FAB>& fine, FabArray<FAB>& crse,
297 const IntVect& ratio, int ngcrse, bool mfiter_is_definitely_safe)
298{
299 AMREX_ASSERT(fine.is_nodal());
300 AMREX_ASSERT(crse.is_nodal());
301 AMREX_ASSERT(crse.nComp() == fine.nComp());
302
303 int ncomp = crse.nComp();
304 using value_type = typename FAB::value_type;
305
306 if (mfiter_is_definitely_safe || isMFIterSafe(fine, crse))
307 {
308#ifdef AMREX_USE_OMP
309#pragma omp parallel if (Gpu::notInLaunchRegion())
310#endif
311 for (MFIter mfi(crse,TilingIfNotGPU()); mfi.isValid(); ++mfi)
312 {
313 const Box& bx = mfi.growntilebox(ngcrse);
314 Array4<value_type> const& crsearr = crse.array(mfi);
315 Array4<value_type const> const& finearr = fine.const_array(mfi);
316
317 AMREX_LAUNCH_HOST_DEVICE_LAMBDA ( bx, tbx,
318 {
319 amrex_avgdown_nodes(tbx,crsearr,finearr,0,0,ncomp,ratio);
320 });
321 }
322 }
323 else
324 {
325 FabArray<FAB> ctmp(amrex::coarsen(fine.boxArray(),ratio), fine.DistributionMap(),
326 ncomp, ngcrse);
327 average_down_nodal(fine, ctmp, ratio, ngcrse);
328 crse.ParallelCopy(ctmp,0,0,ncomp,ngcrse,ngcrse);
329 }
330}

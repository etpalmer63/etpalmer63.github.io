55 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
56 int operator() (const IntVect& iv, int nGrow=0) const noexcept
57 {
58 const auto lo = iv.dim3();
59 int ix_lo = amrex::max((lo.x - nGrow - m_lo.x) / m_bin_size.x - 1, 0);
60 int iy_lo = amrex::max((lo.y - nGrow - m_lo.y) / m_bin_size.y - 1, 0);
61 int iz_lo = amrex::max((lo.z - nGrow - m_lo.z) / m_bin_size.z - 1, 0);
62
63 int ix_hi = amrex::min((lo.x + nGrow - m_lo.x) / m_bin_size.x, m_num_bins.x-1);
64 int iy_hi = amrex::min((lo.y + nGrow - m_lo.y) / m_bin_size.y, m_num_bins.y-1);
65 int iz_hi = amrex::min((lo.z + nGrow - m_lo.z) / m_bin_size.z, m_num_bins.z-1);
66
67 for (int ii = ix_lo; ii <= ix_hi; ++ii) {
68 for (int jj = iy_lo; jj <= iy_hi; ++jj) {
69 for (int kk = iz_lo; kk <= iz_hi; ++kk) {
70 int index = (ii * m_num_bins.y + jj) * m_num_bins.z + kk;
71 for (const auto& nbor : m_bif.getBinIterator(index)) {
72 Box bx = nbor.second;
73 bx.grow(nGrow);
74 if (bx.contains(iv)) return nbor.first;
75 }
76 }
77 }
78 }
79
80 return -1;
81 }

94 {
95 m_defined = true;
96 m_ba = ba;
97 m_geom = geom;
98 int num_boxes = ba.size();
99 m_host_boxes.resize(0);
100 for (int i = 0; i < num_boxes; ++i) m_host_boxes.push_back(ba[i]);
101
102 m_device_boxes.resize(num_boxes);
103 Gpu::copy(Gpu::hostToDevice, m_host_boxes.begin(), m_host_boxes.end(), m_device_boxes.begin());
104
105 // compute the lo, hi and the max box size in each direction
106 ReduceOps<AMREX_D_DECL(ReduceOpMin, ReduceOpMin, ReduceOpMin),
107 AMREX_D_DECL(ReduceOpMax, ReduceOpMax, ReduceOpMax),
108 AMREX_D_DECL(ReduceOpMax, ReduceOpMax, ReduceOpMax)> reduce_op;
109 ReduceData<AMREX_D_DECL(int, int, int),
110 AMREX_D_DECL(int, int, int),
111 AMREX_D_DECL(int, int, int)> reduce_data(reduce_op);
112 using ReduceTuple = typename decltype(reduce_data)::Type;
113
114 const auto boxes_ptr = m_device_boxes.dataPtr();
115 reduce_op.eval(num_boxes, reduce_data,
116 [=] AMREX_GPU_DEVICE (int i) -> ReduceTuple
117 {
118 const Box& box = boxes_ptr[i];
119 IntVect lo = box.smallEnd();
120 IntVect hi = box.bigEnd();
121 IntVect si = box.length();
122 return {AMREX_D_DECL(lo[0], lo[1], lo[2]),
123 AMREX_D_DECL(hi[0], hi[1], hi[2]),
124 AMREX_D_DECL(si[0], si[1], si[2])};
125 });
126
127 ReduceTuple hv = reduce_data.value(reduce_op);
128
129 m_bins_lo = IntVect(AMREX_D_DECL(amrex::get<0>(hv),
130 amrex::get<1>(hv),
131 amrex::get<2>(hv)));
132 m_bins_hi = IntVect(AMREX_D_DECL(amrex::get< AMREX_SPACEDIM >(hv),
133 amrex::get< AMREX_SPACEDIM+1>(hv),
134 amrex::get< AMREX_SPACEDIM+2>(hv)));
135 m_bin_size = IntVect(AMREX_D_DECL(amrex::get<2*AMREX_SPACEDIM>(hv),
136 amrex::get<2*AMREX_SPACEDIM+1>(hv),
137 amrex::get<2*AMREX_SPACEDIM+2>(hv)));
138
139 m_num_bins = (m_bins_hi - m_bins_lo + m_bin_size) / m_bin_size;
140
141 Box bins_box(IntVect::TheZeroVector(), m_num_bins-IntVect::TheUnitVector());
142 IntVect bin_size = m_bin_size;
143 IntVect bins_lo = m_bins_lo;
144 m_bins.build(num_boxes, boxes_ptr, bins_box,
145 [=] AMREX_GPU_DEVICE (const Box& box) noexcept -> IntVect
146 {
147 return (box.smallEnd() - bins_lo) / bin_size;
148 });
149 }

198 template <typename P>
199 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
200 GpuTuple<int, int> operator() (const P& p, int lev_min=-1, int lev_max=-1, int nGrow=0) const noexcept
201 {
202 lev_min = (lev_min == -1) ? 0 : lev_min;
203 lev_max = (lev_max == -1) ? m_size - 1 : lev_max;
204
205 for (int lev = lev_max; lev >= lev_min; --lev)
206 {
207 int grid = m_funcs[lev](p);
208 if (grid >= 0) return makeTuple(grid, lev);
209 }
210
211 for (int lev = lev_min; lev >= lev_min; --lev)
212 {
213 int grid = m_funcs[lev](p, nGrow);
214 if (grid >= 0) return makeTuple(grid, lev);
215 }
216
217 return makeTuple(-1, -1);

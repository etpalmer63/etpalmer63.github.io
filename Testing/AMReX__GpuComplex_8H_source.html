368template <typename T>
369AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
370GpuComplex<T> sqrt (const GpuComplex<T>& a_z) noexcept
371{
372 T x = a_z.real();
373 T y = a_z.imag();
374
375 if (x == T())
376 {
377 T t = std::sqrt(amrex::Math::abs(y) / 2);
378 return GpuComplex<T>(t, y < T() ? -t : t);
379 }
380 else
381 {
382 T t = std::sqrt(2 * (amrex::abs(a_z) + amrex::Math::abs(x)));
383 T u = t / 2;
384 return x > T()
385 ? GpuComplex<T>(u, y / t)
386 : GpuComplex<T>(amrex::Math::abs(y) / t, y < T() ? -u : u);
387 }
388}

174 const auto dxi = geom.InvCellSizeArray();
175 const auto plo = geom.ProbLoArray();

177 const size_t np_real = ptile.numRealParticles();
178 const size_t np_total = vec.size();

181 const auto lo = lbound(bx);
182 const auto hi = ubound(bx);
183 m_bins.build(np_total, pstruct_ptr, bx,
184 [=] AMREX_GPU_DEVICE (const ParticleType& p) noexcept -> IntVect
185 {
186 AMREX_D_TERM(AMREX_ASSERT((p.pos(0)-plo[0])*dxi[0] - lo.x >= 0.0);,
187 AMREX_ASSERT((p.pos(1)-plo[1])*dxi[1] - lo.y >= 0.0);,
188 AMREX_ASSERT((p.pos(2)-plo[2])*dxi[2] - lo.z >= 0.0));
189
190 return IntVect(AMREX_D_DECL(static_cast<int>(amrex::Math::floor((p.pos(0)-plo[0])*dxi[0])) - lo.x,
191 static_cast<int>(amrex::Math::floor((p.pos(1)-plo[1])*dxi[1])) - lo.y,
192 static_cast<int>(amrex::Math::floor((p.pos(2)-plo[2])*dxi[2])) - lo.z));
193 });
194
195 // first pass - count the number of neighbors for each particle
196 m_nbor_counts.resize(np_real+1, 0);
197 m_nbor_offsets.resize(np_real+1);

200 auto pperm = m_bins.permutationPtr();
201 auto poffset = m_bins.offsetsPtr();
202 auto pnbor_offset = m_nbor_offsets.dataPtr();
203
204 AMREX_FOR_1D ( np_real, i,
205 {
206 IntVect iv(AMREX_D_DECL(
207 static_cast<int>(amrex::Math::floor((pstruct_ptr[i].pos(0)-plo[0])*dxi[0])) - lo.x,
208 static_cast<int>(amrex::Math::floor((pstruct_ptr[i].pos(1)-plo[1])*dxi[1])) - lo.y,
209 static_cast<int>(amrex::Math::floor((pstruct_ptr[i].pos(2)-plo[2])*dxi[2])) - lo.z));
210 auto iv3 = iv.dim3();
211
212 int ix = iv3.x;
213 int iy = iv3.y;
214 int iz = iv3.z;
215
216 int nx = hi.x-lo.x+1;
217 int ny = hi.y-lo.y+1;
218 int nz = hi.z-lo.z+1;
219
220 int count = 0;
221 for (int ii = amrex::max(ix-num_cells, 0); ii <= amrex::min(ix+num_cells, nx-1); ++ii) {
222 for (int jj = amrex::max(iy-num_cells, 0); jj <= amrex::min(iy+num_cells, ny-1); ++jj) {
223 for (int kk = amrex::max(iz-num_cells, 0); kk <= amrex::min(iz+num_cells, nz-1); ++kk) {
224 int index = (ii * ny + jj) * nz + kk;
225 for (auto p = poffset[index]; p < poffset[index+1]; ++p) {
226 if (pperm[p] == i) continue;
227 if (call_check_pair(check_pair, pstruct_ptr, i, pperm[p])) {
228 count += 1;
229 }
230 }
231 }
232 }
233 }
234
235 pnbor_counts[i] = count;
236 });
237
238 Gpu::exclusive_scan(m_nbor_counts.begin(), m_nbor_counts.end(), m_nbor_offsets.begin());
239
240 // Now we can allocate and build our neighbor list
241 unsigned int total_nbors;
242#ifdef AMREX_USE_GPU
243 Gpu::dtoh_memcpy(&total_nbors,m_nbor_offsets.dataPtr()+np_real,sizeof(unsigned int));
244#else
245 std::memcpy(&total_nbors,m_nbor_offsets.dataPtr()+np_real,sizeof(unsigned int));
246#endif
247
248 m_nbor_list.resize(total_nbors);
249
250 auto pm_nbor_list = m_nbor_list.dataPtr();
251
252 AMREX_FOR_1D ( np_real, i,
253 {
254 IntVect iv(AMREX_D_DECL(
255 static_cast<int>(amrex::Math::floor((pstruct_ptr[i].pos(0)-plo[0])*dxi[0])) - lo.x,
256 static_cast<int>(amrex::Math::floor((pstruct_ptr[i].pos(1)-plo[1])*dxi[1])) - lo.y,
257 static_cast<int>(amrex::Math::floor((pstruct_ptr[i].pos(2)-plo[2])*dxi[2])) - lo.z));
258 auto iv3 = iv.dim3();
259
260 int ix = iv3.x;
261 int iy = iv3.y;
262 int iz = iv3.z;
263
264 int nx = hi.x-lo.x+1;
265 int ny = hi.y-lo.y+1;
266 int nz = hi.z-lo.z+1;
267
268 int n = 0;
269 for (int ii = amrex::max(ix-num_cells, 0); ii <= amrex::min(ix+num_cells, nx-1); ++ii) {
270 for (int jj = amrex::max(iy-num_cells, 0); jj <= amrex::min(iy+num_cells, ny-1); ++jj) {
271 for (int kk = amrex::max(iz-num_cells, 0); kk <= amrex::min(iz+num_cells, nz-1); ++kk) {
272 int index = (ii * ny + jj) * nz + kk;

274 if (pperm[p] == i) continue;
275 if (call_check_pair(check_pair, pstruct_ptr, i, pperm[p])) {
276 pm_nbor_list[pnbor_offset[i] + n] = pperm[p];
277 ++n;
278 }
279 }
280 }
281 }
282 }
283 });

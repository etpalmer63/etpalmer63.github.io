25AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
26int coarsen (int i, int ratio) noexcept
27{
28 switch (ratio) {
29 case 1: return i;
30 case 2: return (i<0) ? -amrex::Math::abs(i+1)/2 -1 : i/2;
31 case 4: return (i<0) ? -amrex::Math::abs(i+1)/4 -1 : i/4;
32 default: return (i<0) ? -amrex::Math::abs(i+1)/ratio-1 : i/ratio;
33 }
34}

51 class shift_hasher {
52 private:
53 static constexpr unsigned shift1 = sizeof(size_t)>=8 ? 20 : 10;
54 static constexpr unsigned shift2 = sizeof(size_t)>=8 ? 40 : 20;
55 public:
56 std::size_t operator()(const IntVect& vec) const noexcept

58 AMREX_D_DECL(std::size_t ret0 = vec[0], ret1 = vec[1], ret2 = vec[2]);
59#if AMREX_SPACEDIM == 1
60 return ret0;
61#elif AMREX_SPACEDIM == 2
62 return ret0 ^ (ret1 << shift1);
63#else
64 return ret0 ^ (ret1 << shift1) ^ (ret2 << shift2);
65#endif
66 }
67 };

558AMREX_GPU_HOST_DEVICE
559AMREX_FORCE_INLINE
560IntVect&
561IntVect::coarsen (int s) noexcept
562{
563 BL_ASSERT(s > 0);
564 switch (s) {
565 case 1:
566 break;
567 case 2:
568 AMREX_D_TERM(vect[0] = (vect[0]<0) ? -amrex::Math::abs(vect[0]+1)/2-1 : vect[0]/2;,
569 vect[1] = (vect[1]<0) ? -amrex::Math::abs(vect[1]+1)/2-1 : vect[1]/2;,
570 vect[2] = (vect[2]<0) ? -amrex::Math::abs(vect[2]+1)/2-1 : vect[2]/2;);
571 break;
572 case 4:
573 AMREX_D_TERM(vect[0] = (vect[0]<0) ? -amrex::Math::abs(vect[0]+1)/4-1 : vect[0]/4;,
574 vect[1] = (vect[1]<0) ? -amrex::Math::abs(vect[1]+1)/4-1 : vect[1]/4;,
575 vect[2] = (vect[2]<0) ? -amrex::Math::abs(vect[2]+1)/4-1 : vect[2]/4;);
576 break;
577 default:
578 AMREX_D_TERM(vect[0] = (vect[0]<0) ? -amrex::Math::abs(vect[0]+1)/s-1 : vect[0]/s;,
579 vect[1] = (vect[1]<0) ? -amrex::Math::abs(vect[1]+1)/s-1 : vect[1]/s;,
580 vect[2] = (vect[2]<0) ? -amrex::Math::abs(vect[2]+1)/s-1 : vect[2]/s;);
581 }
582 return *this;
583}

598AMREX_FORCE_INLINE
599int
600IntVect::maxDir(bool a_doAbsValue) const noexcept
601{
602 int retval = 0;
603 if(a_doAbsValue)
604 {
605 int maxval = abs((*this)[0]);
606 for(int idir = 1; idir < SpaceDim; idir++)
607 {
608 int curval = amrex::Math::abs((*this)[idir]);
609 if(curval > maxval)
610 {
611 maxval = curval;
612 retval = idir;
613 }
614 }
615 }
616 else
617 {
618 int maxval = (*this)[0];
619 for(int idir = 1; idir < SpaceDim; idir++)
620 {
621 int curval = (*this)[idir];
622 if(curval > maxval)
623 {
624 maxval = curval;
625 retval = idir;
626 }
627 }
628 }
629 return retval;
630}

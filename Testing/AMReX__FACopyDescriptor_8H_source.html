239template <class FAB>
240void
241FabArrayCopyDescriptor<FAB>::AddBoxDoIt (FabArrayId fabarrayid,
242 const Box& destFabBox,
243 BoxList* returnedUnfilledBoxes,
244 int faindex,
245 int srccomp,
246 int destcomp,
247 int numcomp,
248 bool bUseValidBox,
249 BoxDomain& unfilledBoxDomain)
250{
251 const int myProc = ParallelDescriptor::MyProc();
252
253 FabArray<FAB>* fabArray = fabArrays[fabarrayid.Id()];
254
255 BL_ASSERT(faindex >= 0 && faindex < fabArray->size());
256
257 Box intersect = destFabBox;
258
259 if (bUseValidBox)
260 {
261 intersect &= fabArray->box(faindex);
262 }
263 else
264 {
265 intersect &= fabArray->fabbox(faindex);
266 }
267
268 if (intersect.ok())
269 {
270 FabCopyDescriptor<FAB>* fcd = new FabCopyDescriptor<FAB>;
271
272 int remoteProc = fabArray->DistributionMap()[faindex];
273 if(remoteProc >= ParallelDescriptor::NProcs()) {
274 amrex::Abort("Bad remoteProc: "
275 + std::to_string(ParallelDescriptor::MyProc())
276 + ":: _in AddBoxDoIt: nProcs remoteProc = "
277 + std::to_string(ParallelDescriptor::NProcs())
278 + " " + std::to_string(remoteProc) + "\n");
279 }
280 fcd->fillBoxId = nextFillBoxId;
281 fcd->subBox = intersect;
282 fcd->myProc = myProc;
283 fcd->copyFromProc = remoteProc;
284 fcd->copyFromIndex = faindex;
285 fcd->srcComp = srccomp;
286 fcd->destComp = destcomp;
287 fcd->nComp = numcomp;
288
289 if (ParallelDescriptor::sameTeam(remoteProc))
290 {
291 //
292 // Data is local.
293 //
294 fcd->fillType = FillLocally;
295 fcd->localFabSource = &(*fabArray)[faindex];
296 }
297 else
298 {
299 //
300 // Data is remote.
301 //
302 FabArrayBase::FabComTag fabComTag;
303
304 dataAvailable = false;
305 fcd->fillType = FillRemotely;
306 fcd->localFabSource = new FAB(intersect, numcomp);
307 fcd->cacheDataAllocated = true;
308 fabComTag.fabArrayId = fabarrayid.Id();
309 fabComTag.fillBoxId = nextFillBoxId;
310 fabComTag.fabIndex = faindex;
311 fabComTag.procThatNeedsData = myProc;
312 fabComTag.procThatHasData = remoteProc;
313 fabComTag.box = intersect;
314 fabComTag.srcComp = srccomp;
315 fabComTag.destComp = destcomp;
316 fabComTag.nComp = numcomp;
317 //
318 // Do not send the data yet.
319 //
320 fabComTagList.push_back(fabComTag);
321 }
322
323 fabCopyDescList[fabarrayid.Id()].insert(FCDMapValueType(fcd->fillBoxId,fcd));
324
325 if (returnedUnfilledBoxes != 0)
326 {
327 unfilledBoxDomain.rmBox(intersect);
328 }
329 }
330}

332template <class FAB>
333FillBoxId
334FabArrayCopyDescriptor<FAB>::AddBox (FabArrayId fabarrayid,
335 const Box& destFabBox,
336 BoxList* returnedUnfilledBoxes,
337 int srccomp,
338 int destcomp,
339 int numcomp)
340{
341 BoxDomain unfilledBoxDomain(destFabBox.ixType());
342
343 if (returnedUnfilledBoxes != 0)
344 {
345 unfilledBoxDomain.add(destFabBox);
346 }
347
348 std::vector< std::pair<int,Box> > isects;
349
350 fabArrays[fabarrayid.Id()]->boxArray().intersections(destFabBox,isects);
351
352 for (int j = 0, N = isects.size(); j < N; j++)
353 {
354 AddBoxDoIt(fabarrayid,
355 destFabBox,
356 returnedUnfilledBoxes,
357 isects[j].first,
358 srccomp,
359 destcomp,
360 numcomp,
361 true,
362 unfilledBoxDomain);
363 }
364
365 if (returnedUnfilledBoxes != 0)
366 {
367 returnedUnfilledBoxes->clear();
368 (*returnedUnfilledBoxes) = unfilledBoxDomain.boxList();
369 }
370
371 return FillBoxId(nextFillBoxId++, destFabBox);
372}

454template <class FAB>
455void
456FabArrayCopyDescriptor<FAB>::CollectData ()
457{
458 dataAvailable = true;
459
460 if (ParallelDescriptor::NProcs() == 1) return;
461
462#ifdef BL_USE_MPI
463 typedef typename FAB::value_type value_type;
464
465 BL_PROFILE("FabArrayCopyDescriptor::CollectData()");
466
467 const int MyProc = ParallelDescriptor::MyProc();
468 amrex::ignore_unused(MyProc);
469
470 int Total_Rcvs_Size = 0;
471 //
472 // We use this to make finding matching FabComTags more efficient.
473 //
474 std::map< int,FabComTagIterContainer > RcvTags;
475
476 std::map<int,int> Snds, Rcvs, Npts;
477 //
478 // Set Rcvs[i] to # of blocks needed from CPU i
479 //
480 for (FabComTagContainer::const_iterator it = fabComTagList.begin(),
481 End = fabComTagList.end();
482 it != End;
483 ++it)
484 {
485 BL_ASSERT(it->box.ok());
486 BL_ASSERT(it->procThatNeedsData == MyProc);
487 BL_ASSERT(it->procThatHasData != MyProc);
488
489 const int Who = it->procThatHasData;
490 const int Cnt = (it->box.numPts())*(it->nComp);
491
492 RcvTags[Who].push_back(it);
493
494 Total_Rcvs_Size += Cnt;
495
496 if (Rcvs.count(Who) > 0)
497 {
498 Rcvs[Who] += 1;
499 }
500 else
501 {
502 Rcvs[Who] = 1;
503 }
504
505 if (Npts.count(Who) > 0)
506 {
507 Npts[Who] += Cnt;
508 }
509 else
510 {
511 Npts[Who] = Cnt;
512 }
513 }
514 BL_ASSERT(Rcvs.count(MyProc) == 0);
515
516 BL_ASSERT((Total_Rcvs_Size*sizeof(value_type))
517 < static_cast<std::size_t>(std::numeric_limits<int>::max()));
518
519 const int NProcs = ParallelDescriptor::NProcs();
520
521 {
522 Vector<int> SndsArray(NProcs,0), RcvsArray(NProcs,0);
523
524 for (std::map<int,int>::const_iterator it = Rcvs.begin(), End = Rcvs.end(); it != End; ++it)
525 {
526 RcvsArray[it->first] = it->second;
527 }
528
529 {
530 BL_PROFILE_VAR("CollectData_Alltoall()", blpvCDATA);
531 BL_COMM_PROFILE(BLProfiler::Alltoall, sizeof(int), ParallelDescriptor::MyProc(),
532 BLProfiler::BeforeCall());
533
534 BL_MPI_REQUIRE( MPI_Alltoall(RcvsArray.dataPtr(),
535 1,
536 ParallelDescriptor::Mpi_typemap<int>::type(),
537 SndsArray.dataPtr(),
538 1,
539 ParallelDescriptor::Mpi_typemap<int>::type(),
540 ParallelDescriptor::Communicator()) );
541
542 BL_COMM_PROFILE(BLProfiler::Alltoall, sizeof(int), ParallelDescriptor::MyProc(),
543 BLProfiler::AfterCall());
544
545 BL_PROFILE_VAR_STOP(blpvCDATA);
546 }
547 BL_ASSERT(SndsArray[MyProc] == 0);
548
549 for (int i = 0; i < NProcs; i++)
550 if (SndsArray[i] > 0)
551 Snds[i] = SndsArray[i];
552 }
553
554 // There are two rounds of send and recv.
555 // First, the data receivers need to send the data senders meta-data (e.g., boxes).
556 // Then, the senders know what data to send and perform send.
557 const int SeqNum_md = ParallelDescriptor::SeqNum();
558 const int SeqNum_data = ParallelDescriptor::SeqNum();
559
560 const int N_snds = Snds.size();
561 const int N_rcvs = Rcvs.size();
562
563 if ( N_snds == 0 && N_rcvs == 0 ) return;
564
565 const int Nints = 4 + 3*AMREX_SPACEDIM; // # of ints in a meta-data
566
567 // for meta-data
568 Vector<int> md_sender, md_offset, md_icnts, md_bcnts;
569 int* md_recv_data = nullptr;
570 Vector<int*> md_send_data;
571 Vector<MPI_Request> md_recv_reqs, md_send_reqs;
572
573 // for data
574 Vector<int> data_sender, data_offset;
575 value_type* recv_data = nullptr;
576 Vector<value_type*> send_data;
577 Vector<MPI_Request> data_recv_reqs, data_send_reqs;
578
579 if (N_snds > 0)
580 {
581 // Recv meta-data
582
583 int N = 0;
584 for (std::map<int,int>::const_iterator it = Snds.begin(), End = Snds.end(); it != End; ++it)
585 {
586 md_sender.push_back(it->first);
587 md_bcnts.push_back(it->second);
588 int cnt = it->second * Nints;
589 md_icnts.push_back(cnt);
590 md_offset.push_back(N);
591 N += cnt;
592 }
593
594 md_recv_data = static_cast<int*>(amrex::The_Arena()->alloc(N*sizeof(int)));
595
596 for (int i = 0; i < N_snds; ++i)
597 {
598 md_recv_reqs.push_back(ParallelDescriptor::Arecv(&md_recv_data[md_offset[i]],
599 md_icnts[i], md_sender[i],
600 SeqNum_md).req());
601 }
602 }
603
604 if (N_rcvs > 0)
605 {
606 // Send meta-data
607 for (std::map<int,int>::const_iterator it = Rcvs.begin(), End = Rcvs.end(); it != End; ++it)
608 {
609 int rank = it->first;
610 int Nmds = it->second;
611 int cnt = Nmds * Nints;
612
613 int* p = static_cast<int*>(amrex::The_Arena()->alloc(cnt*sizeof(int)));
614 md_send_data.push_back(p);
615
616 const FabComTagIterContainer& tags = RcvTags[rank];
617
618 // initialized the data
619 int * md = p;
620 for (int i = 0; i < Nmds; ++i, md += Nints)
621 {
622 md[0] = tags[i]->fabArrayId;
623 md[1] = tags[i]->fabIndex;
624 md[2] = tags[i]->srcComp;
625 md[3] = tags[i]->nComp;
626 const int* lo = tags[i]->box.loVect();
627 const int* hi = tags[i]->box.hiVect();
628 const IntVect& bxtyp = tags[i]->box.type();
629 const int* tp = bxtyp.getVect();
630 AMREX_D_EXPR(md[4] = lo[0],
631 md[5] = lo[1],
632 md[6] = lo[2]);
633 AMREX_D_EXPR(md[4+ AMREX_SPACEDIM] = hi[0],
634 md[5+ AMREX_SPACEDIM] = hi[1],
635 md[6+ AMREX_SPACEDIM] = hi[2]);
636 AMREX_D_EXPR(md[4+2*AMREX_SPACEDIM] = tp[0],
637 md[5+2*AMREX_SPACEDIM] = tp[1],
638 md[6+2*AMREX_SPACEDIM] = tp[2]);
639 }
640
641 md_send_reqs.push_back(ParallelDescriptor::Asend(p,cnt,rank,SeqNum_md).req());
642 }
643 }
644
645 if (N_rcvs > 0)
646 {
647 recv_data = static_cast<value_type*>(amrex::The_Arena()->alloc(Total_Rcvs_Size*sizeof(value_type)));
648
649 // Post receives for data
650 int Idx = 0;
651 for (std::map<int,int>::const_iterator it = Npts.begin(); it != Npts.end(); ++it)
652 {
653 int Who = it->first;
654 int Cnt = it->second;
655 BL_ASSERT(Cnt > 0);
656 BL_ASSERT(Cnt < std::numeric_limits<int>::max());
657 data_sender.push_back(Who);
658 data_recv_reqs.push_back(ParallelDescriptor::Arecv(&recv_data[Idx],
659 Cnt,Who,SeqNum_data).req());
660 data_offset.push_back(Idx);
661 Idx += Cnt;
662 }
663 }
664
665 // Wait on meta-data and do send
666 if (N_snds > 0)
667 {
668 int send_counter = 0;
669 while (send_counter++ < N_snds)
670 {
671 MPI_Status status;
672 int index;
673 ParallelDescriptor::Waitany(md_recv_reqs, index, status);
674
675 int rank = status.MPI_SOURCE;
676 BL_ASSERT(status.MPI_TAG == SeqNum_md);
677 BL_ASSERT(rank == md_sender[index]);
678
679 const int* p = &md_recv_data[md_offset[index]];
680 int numboxes = md_bcnts[index];
681 Vector<int> faid(numboxes);
682 Vector<int> fidx(numboxes);
683 Vector<int> scomp(numboxes);
684 Vector<int> ncomp(numboxes);
685 Vector<int> npts(numboxes);
686 Vector<Box> bxs;
687 int N = 0;
688 const int * md = p;
689 for (int i = 0; i < numboxes; ++i, md += Nints)
690 {
691 faid[i] = md[0];
692 fidx[i] = md[1];
693 scomp[i] = md[2];
694 ncomp[i] = md[3];
695 bxs.push_back(Box(IntVect(&md[4]),
696 IntVect(&md[4+AMREX_SPACEDIM]),
697 IntVect(&md[4+AMREX_SPACEDIM*2])));
698 npts[i] = bxs.back().numPts()*ncomp[i];
699 N += npts[i];
700 }
701
702 BL_ASSERT(N < std::numeric_limits<int>::max());
703
704 value_type* data = static_cast<value_type*>(amrex::The_Arena()->alloc(N*sizeof(value_type)));
705 value_type* dptr = data;
706 send_data.push_back(data);
707
708 for (int i = 0; i < numboxes; ++i)
709 {
710 (*fabArrays[faid[i]])[fidx[i]].template copyToMem<RunOn::Host>(bxs[i],scomp[i],ncomp[i],dptr);
711 dptr += npts[i];
712 }
713
714 data_send_reqs.push_back(ParallelDescriptor::Asend(data,N,rank,SeqNum_data).req());
715 }
716
717 amrex::The_Arena()->free(md_recv_data);
718 }
719
720 // Wait and upack data
721 if (N_rcvs > 0)
722 {
723 Vector<MPI_Status> stats(N_rcvs);
724
725 ParallelDescriptor::Waitall(md_send_reqs, stats);
726 for (int i = 0; i < N_rcvs; ++i) {
727 amrex::The_Arena()->free(md_send_data[i]);
728 }
729
730 ParallelDescriptor::Waitall(data_recv_reqs, stats);
731
732 std::pair<FCDMapIter,FCDMapIter> match;
733 std::map< int,FabComTagIterContainer >::const_iterator found;
734
735 for (int k = 0; k < N_rcvs; k++)
736 {
737 const int Who = data_sender[k];
738 const value_type* dptr = &recv_data[data_offset[k]];
739
740 BL_ASSERT(dptr != 0);
741
742 found = RcvTags.find(Who);
743
744 BL_ASSERT(found != RcvTags.end());
745
746 const FabComTagIterContainer& tags = found->second;
747
748 for (FabComTagIterContainer::const_iterator it = tags.begin(), End = tags.end();
749 it != End;
750 ++it)
751 {
752 const FabArrayBase::FabComTag& tag = **it;
753
754 BL_ASSERT(tag.procThatHasData == Who);
755
756 match = fabCopyDescList[tag.fabArrayId].equal_range(tag.fillBoxId);
757
758 for (FCDMapIter fmi = match.first; fmi != match.second; ++fmi)
759 {
760 FabCopyDescriptor<FAB>* fcdp = (*fmi).second;
761
762 BL_ASSERT(fcdp->fillBoxId == tag.fillBoxId);
763
764 if (fcdp->subBox == tag.box)
765 {
766 BL_ASSERT(fcdp->localFabSource->dataPtr() != 0);
767 BL_ASSERT(fcdp->localFabSource->box() == tag.box);
768 const int Cnt = tag.box.numPts()*tag.nComp;
769 fcdp->localFabSource->template copyFromMem<RunOn::Host>(tag.box,0,tag.nComp,dptr);
770 dptr += Cnt;
771 break;
772 }
773 }
774 }
775 }
776
777 amrex::The_Arena()->free(recv_data);
778 }
779
780 // Finished send
781 if (N_snds > 0)
782 {
783 Vector<MPI_Status> stats(N_snds);
784 ParallelDescriptor::Waitall(data_send_reqs, stats);
785
786 for (int i = 0; i < N_snds; ++i) {
787 amrex::The_Arena()->free(send_data[i]);
788 }
789 }
790
791#endif /*BL_USE_MPI*/
792}

795template <class FAB>
796void
797FabArrayCopyDescriptor<FAB>::FillFab (FabArrayId faid,
798 const FillBoxId& fillboxid,
799 FAB& destFab)
800{
801 BL_ASSERT(dataAvailable);
802
803 std::pair<FCDMapIter,FCDMapIter> match = fabCopyDescList[faid.Id()].equal_range(fillboxid.Id());
804
805 for (FCDMapIter fmi = match.first; fmi != match.second; ++fmi)
806 {
807 FabCopyDescriptor<FAB>* fcdp = (*fmi).second;
808
809 BL_ASSERT(fcdp->fillBoxId == fillboxid.Id());
810
811 destFab.template copy<RunOn::Host>
812 (*fcdp->localFabSource,

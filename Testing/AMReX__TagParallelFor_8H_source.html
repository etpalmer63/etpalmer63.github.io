96AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
97std::enable_if_t<std::is_same<std::decay_t<decltype(std::declval<T>().box())>, Box>::value>
98tagparfor_call_f (
99#ifdef AMREX_USE_DPCPP
100 sycl::nd_item<1> const& item,
101#endif
102 int icell, T const& tag, F&& f) noexcept
103{
104 int ncells = tag.box().numPts();
105 const auto len = amrex::length(tag.box());
106 const auto lo = amrex::lbound(tag.box());
107 int k = icell / (len.x*len.y);
108 int j = (icell - k*(len.x*len.y)) / len.x;
109 int i = (icell - k*(len.x*len.y)) - j*len.x;
110 i += lo.x;
111 j += lo.y;
112 k += lo.z;
113#ifdef AMREX_USE_DPCPP
114 f(item, icell, ncells, i, j, k, tag);
115#else
116 f( icell, ncells, i, j, k, tag);
117#endif
118}

137template <class TagType, class F>
138void
139ParallelFor_doit (Vector<TagType> const& tags, F && f)
140{
141 const int ntags = tags.size();
142 if (ntags == 0) return;
143
144 int ntotwarps = 0;
145 Vector<int> nwarps;
146 nwarps.reserve(ntags+1);
147 for (int i = 0; i < ntags; ++i)
148 {
149 auto& tag = tags[i];
150 nwarps.push_back(ntotwarps);
151 ntotwarps += static_cast<int>((get_tag_size(tag)
152 + Gpu::Device::warp_size-1) / Gpu::Device::warp_size);
153 }
154 nwarps.push_back(ntotwarps);
155
156 std::size_t sizeof_tags = ntags*sizeof(TagType);
157 std::size_t offset_nwarps = Arena::align(sizeof_tags);
158 std::size_t sizeof_nwarps = (ntags+1)*sizeof(int);
159 std::size_t total_buf_size = offset_nwarps + sizeof_nwarps;
160
161 char* h_buffer = (char*)The_Pinned_Arena()->alloc(total_buf_size);
162 char* d_buffer = (char*)The_Arena()->alloc(total_buf_size);
163
164 std::memcpy(h_buffer, tags.data(), sizeof_tags);
165 std::memcpy(h_buffer+offset_nwarps, nwarps.data(), sizeof_nwarps);
166 Gpu::htod_memcpy_async(d_buffer, h_buffer, total_buf_size);
167
168 auto d_tags = reinterpret_cast<TagType*>(d_buffer);
169 auto d_nwarps = reinterpret_cast<int*>(d_buffer+offset_nwarps);
170
171 constexpr int nthreads = 256;
172 constexpr int nwarps_per_block = nthreads/Gpu::Device::warp_size;
173 int nblocks = (ntotwarps + nwarps_per_block-1) / nwarps_per_block;
174
175 amrex::launch(nblocks, nthreads, Gpu::gpuStream(),
176#ifdef AMREX_USE_DPCPP
177 [=] AMREX_GPU_DEVICE (sycl::nd_item<1> const& item) noexcept
178 AMREX_REQUIRE_SUBGROUP_SIZE(Gpu::Device::warp_size)
179#else
180 [=] AMREX_GPU_DEVICE () noexcept
181#endif
182 {
183#ifdef AMREX_USE_DPCPP
184 int g_tid = item.get_global_id(0);
185#else
186 int g_tid = blockDim.x*blockIdx.x + threadIdx.x;
187#endif
188 int g_wid = g_tid / Gpu::Device::warp_size;
189 if (g_wid >= ntotwarps) return;
190
191 int tag_id = amrex::bisect(d_nwarps, 0, ntags, g_wid);
192
193 int b_wid = g_wid - d_nwarps[tag_id]; // b_wid'th warp on this box
194#ifdef AMREX_USE_DPCPP
195 int lane = item.get_local_id(0) % Gpu::Device::warp_size;
196#else
197 int lane = threadIdx.x % Gpu::Device::warp_size;
198#endif
199 int icell = b_wid*Gpu::Device::warp_size + lane;
200
201#ifdef AMREX_USE_DPCPP
202 tagparfor_call_f(item, icell, d_tags[tag_id], f);
203#else
204 tagparfor_call_f( icell, d_tags[tag_id], f);
205#endif
206 });
207
208 Gpu::synchronize();
209 The_Pinned_Arena()->free(h_buffer);
210 The_Arena()->free(d_buffer);
211}

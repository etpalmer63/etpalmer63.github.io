349 template <typename I, typename MF, typename D, typename F>
350 void eval_mf (I, MF const& mf, IntVect const& nghost, int ncomp, D& reduce_data, F&&f)
351 {
352 using ReduceTuple = typename D::Type;
353 const int nboxes = mf.local_size();
354 if (nboxes > 0) {
355 auto const& parforinfo = mf.getParForInfo(nghost,AMREX_GPU_MAX_THREADS);
356 auto par_for_blocks = parforinfo.getBlocks();
357 const int nblocks = par_for_blocks.first[nboxes];
358 const int block_0_size = par_for_blocks.first[1];
359 const int* dp_nblocks = par_for_blocks.second;
360 const Box* dp_boxes = parforinfo.getBoxes();
361
362 auto const& stream = Gpu::gpuStream();
363 auto pdst = reduce_data.devicePtr(stream);
364 int nblocks_ec = std::min(nblocks, reduce_data.maxBlocks());
365 reduce_data.nBlocks(stream) = nblocks_ec;
366
367#ifdef AMREX_USE_DPCPP
368 // device reduce needs local(i.e., shared) memory
369 constexpr std::size_t shared_mem_bytes = sizeof(unsigned long long)*Gpu::Device::warp_size;
370 amrex::launch(nblocks_ec, AMREX_GPU_MAX_THREADS, shared_mem_bytes, stream,
371 [=] AMREX_GPU_DEVICE (Gpu::Handler const& gh) noexcept
372 {
373 Dim1 blockIdx {gh.blockIdx()};
374 Dim1 threadIdx{gh.threadIdx()};
375#else
376 amrex::launch_global<AMREX_GPU_MAX_THREADS>
377 <<<nblocks_ec, AMREX_GPU_MAX_THREADS, 0, stream>>>
378 ([=] AMREX_GPU_DEVICE () noexcept
379 {
380#endif
381 ReduceTuple r;
382 Reduce::detail::for_each_init<0, ReduceTuple, Ps...>(r);
383 ReduceTuple& dst = pdst[blockIdx.x];
384 if (threadIdx.x == 0) {
385 dst = r;
386 }
387 for (int iblock = blockIdx.x; iblock < nblocks; iblock += nblocks_ec) {
388 int ibox, icell;
389 if (dp_nblocks) {
390 ibox = amrex::bisect(dp_nblocks, 0, nboxes, iblock);
391 icell = (iblock-dp_nblocks[ibox])*AMREX_GPU_MAX_THREADS + threadIdx.x;
392 } else {
393 ibox = iblock / block_0_size;
394 icell = (iblock-ibox*block_0_size)*AMREX_GPU_MAX_THREADS + threadIdx.x;
395 }
396
397 Box const& b = dp_boxes[ibox];
398 int ncells = b.numPts();
399 if (icell < ncells) {
400 const auto len = amrex::length(b);
401 int k = icell / (len.x*len.y);
402 int j = (icell - k*(len.x*len.y)) / len.x;
403 int i = (icell - k*(len.x*len.y)) - j*len.x;
404 AMREX_D_TERM(i += b.smallEnd(0);,
405 j += b.smallEnd(1);,
406 k += b.smallEnd(2););
407 Reduce::detail::mf_call_f<I, F, ReduceTuple, Ps...>
408 (f, ibox, i, j, k, ncomp, r);
409 }
410 }
411#ifdef AMREX_USE_DPCPP
412 Reduce::detail::for_each_parallel<0, ReduceTuple, Ps...>(dst, r, gh);
413#else
414 Reduce::detail::for_each_parallel<0, ReduceTuple, Ps...>(dst, r);
415#endif
416 });
417 }

478 template <typename D, typename F>
479 void eval (Box const& box, D & reduce_data, F&& f)
480 {
481 using ReduceTuple = typename D::Type;
482 auto const& stream = Gpu::gpuStream();
483 auto dp = reduce_data.devicePtr(stream);
484 int& nblocks = reduce_data.nBlocks(stream);
485 int ncells = box.numPts();
486 const auto lo = amrex::lbound(box);
487 const auto len = amrex::length(box);
488 const auto lenxy = len.x*len.y;
489 const auto lenx = len.x;
490 IndexType ixtype = box.ixType();
491 constexpr int nitems_per_thread = 4;
492 int nblocks_ec = (ncells + nitems_per_thread*AMREX_GPU_MAX_THREADS-1)
493 / (nitems_per_thread*AMREX_GPU_MAX_THREADS);
494 nblocks_ec = std::min(nblocks_ec, reduce_data.maxBlocks());
495#ifdef AMREX_USE_DPCPP
496 // device reduce needs local(i.e., shared) memory
497 constexpr std::size_t shared_mem_bytes = sizeof(unsigned long long)*Gpu::Device::warp_size;
498 amrex::launch(nblocks_ec, AMREX_GPU_MAX_THREADS, shared_mem_bytes, stream,
499 [=] AMREX_GPU_DEVICE (Gpu::Handler const& gh) noexcept
500 {
501 Dim1 blockIdx {gh.blockIdx()};
502 Dim1 threadIdx{gh.threadIdx()};
503 Dim1 blockDim {gh.blockDim()};
504 Dim1 gridDim {gh.gridDim()};
505#else
506 amrex::launch(nblocks_ec, AMREX_GPU_MAX_THREADS, 0, stream,
507 [=] AMREX_GPU_DEVICE () noexcept
508 {
509#endif
510 ReduceTuple r;
511 Reduce::detail::for_each_init<0, ReduceTuple, Ps...>(r);
512 ReduceTuple& dst = *(dp+blockIdx.x);
513 if (threadIdx.x == 0 && static_cast<int>(blockIdx.x) >= nblocks) {
514 dst = r;
515 }
516 for (int icell = blockDim.x*blockIdx.x+threadIdx.x, stride = blockDim.x*gridDim.x;
517 icell < ncells; icell += stride) {
518 int k = icell / lenxy;
519 int j = (icell - k*lenxy) / lenx;
520 int i = (icell - k*lenxy) - j*lenx;
521 i += lo.x;
522 j += lo.y;
523 k += lo.z;
524 auto pr = Reduce::detail::call_f(f,i,j,k,ixtype);
525 Reduce::detail::for_each_local<0, ReduceTuple, Ps...>(r, pr);
526 }
527#ifdef AMREX_USE_DPCPP
528 Reduce::detail::for_each_parallel<0, ReduceTuple, Ps...>(dst, r, gh);
529#else
530 Reduce::detail::for_each_parallel<0, ReduceTuple, Ps...>(dst, r);
531#endif
532 });
533 nblocks = std::max(nblocks, nblocks_ec);
534 }

597 typename M=std::enable_if_t<std::is_integral<N>::value> >
598 void eval (N n, D & reduce_data, F&& f)
599 {
600 if (n <= 0) return;
601 using ReduceTuple = typename D::Type;
602 auto const& stream = Gpu::gpuStream();
603 auto dp = reduce_data.devicePtr(stream);
604 int& nblocks = reduce_data.nBlocks(stream);
605 constexpr int nitems_per_thread = 4;
606 int nblocks_ec = (n + nitems_per_thread*AMREX_GPU_MAX_THREADS-1)
607 / (nitems_per_thread*AMREX_GPU_MAX_THREADS);
608 nblocks_ec = std::min(nblocks_ec, reduce_data.maxBlocks());
609#ifdef AMREX_USE_DPCPP
610 // device reduce needs local(i.e., shared) memory
611 constexpr std::size_t shared_mem_bytes = sizeof(unsigned long long)*Gpu::Device::warp_size;
612 amrex::launch(nblocks_ec, AMREX_GPU_MAX_THREADS, shared_mem_bytes, stream,
613 [=] AMREX_GPU_DEVICE (Gpu::Handler const& gh) noexcept
614 {
615 Dim1 blockIdx {gh.blockIdx()};
616 Dim1 threadIdx{gh.threadIdx()};
617 Dim1 blockDim {gh.blockDim()};
618 Dim1 gridDim {gh.gridDim()};
619#else
620 amrex::launch(nblocks_ec, AMREX_GPU_MAX_THREADS, 0, stream,
621 [=] AMREX_GPU_DEVICE () noexcept
622 {
623#endif
624 ReduceTuple r;
625 Reduce::detail::for_each_init<0, ReduceTuple, Ps...>(r);
626 ReduceTuple& dst = *(dp+blockIdx.x);
627 if (threadIdx.x == 0 && static_cast<int>(blockIdx.x) >= nblocks) {
628 dst = r;
629 }
630 for (N i = blockDim.x*blockIdx.x+threadIdx.x, stride = blockDim.x*gridDim.x;
631 i < n; i += stride) {
632 auto pr = f(i);
633 Reduce::detail::for_each_local<0, ReduceTuple, Ps...>(r,pr);
634 }
635#ifdef AMREX_USE_DPCPP
636 Reduce::detail::for_each_parallel<0, ReduceTuple, Ps...>(dst, r, gh);
637#else
638 Reduce::detail::for_each_parallel<0, ReduceTuple, Ps...>(dst, r);
639#endif
640 });
641 nblocks = amrex::max(nblocks, nblocks_ec);
642 }

647 using ReduceTuple = typename D::Type;
648 auto const& stream = Gpu::gpuStream();
649 auto hp = reduce_data.hostPtr();
650 auto dp = reduce_data.devicePtr();
651 auto const& nblocks = reduce_data.nBlocks();
652 int maxblocks = reduce_data.maxBlocks();
653#ifdef AMREX_USE_DPCPP
654 // device reduce needs local(i.e., shared) memory
655 constexpr std::size_t shared_mem_bytes = sizeof(unsigned long long)*Gpu::Device::warp_size;
656#ifndef AMREX_NO_DPCPP_REDUCE_WORKAROUND
657 // xxxxx DPCPP todo: reduce bug workaround
658 Gpu::DeviceVector<ReduceTuple> dtmp(1);
659 auto presult = dtmp.data();
660#else
661 auto presult = hp;
662#endif

664 [=] AMREX_GPU_DEVICE (Gpu::Handler const& gh) noexcept
665 {
666 ReduceTuple r;
667 Reduce::detail::for_each_init<0, ReduceTuple, Ps...>(r);
668 ReduceTuple dst = r;
669 for (int istream = 0, nstreams = nblocks.size(); istream < nstreams; ++istream) {
670 auto dp_stream = dp+istream*maxblocks;
671 for (int i = gh.item->get_global_id(0), stride = gh.item->get_global_range(0);
672 i < nblocks[istream]; i += stride) {
673 Reduce::detail::for_each_local<0, ReduceTuple, Ps...>(r, dp_stream[i]);
674 }
675 }

677 if (gh.threadIdx() == 0) { *presult = dst; }
678 });
679#ifndef AMREX_NO_DPCPP_REDUCE_WORKAROUND
680 Gpu::dtoh_memcpy_async(hp, dtmp.data(), sizeof(ReduceTuple));
681#endif
682#else
683 amrex::launch(1, AMREX_GPU_MAX_THREADS, 0, stream,
684 [=] AMREX_GPU_DEVICE () noexcept
685 {
686 ReduceTuple r;
687 Reduce::detail::for_each_init<0, ReduceTuple, Ps...>(r);
688 ReduceTuple dst = r;
689 for (int istream = 0, nstreams = nblocks.size(); istream < nstreams; ++istream) {

811 ec.numBlocks.x = std::min(ec.numBlocks.x, Gpu::Device::maxBlocksPerLaunch());

835 });

840 if (threadIdx.x == 0) has_any = *dp;

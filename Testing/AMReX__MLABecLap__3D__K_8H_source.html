199AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
200void abec_gsrb (int i, int j, int k, int n, Array4<Real> const& phi, Array4<Real const> const& rhs,
201 Real alpha, Array4<Real const> const& a,
202 Real dhx, Real dhy, Real dhz,
203 Array4<Real const> const& bX, Array4<Real const> const& bY,
204 Array4<Real const> const& bZ,
205 Array4<int const> const& m0, Array4<int const> const& m2,
206 Array4<int const> const& m4,
207 Array4<int const> const& m1, Array4<int const> const& m3,
208 Array4<int const> const& m5,
209 Array4<Real const> const& f0, Array4<Real const> const& f2,
210 Array4<Real const> const& f4,
211 Array4<Real const> const& f1, Array4<Real const> const& f3,
212 Array4<Real const> const& f5,
213 Box const& vbox, int redblack) noexcept
214{
215 constexpr Real omega = Real(1.15);
216
217 if ((i+j+k+redblack)%2 == 0) {
218 const auto vlo = amrex::lbound(vbox);
219 const auto vhi = amrex::ubound(vbox);
220
221 Real cf0 = (i == vlo.x && m0(vlo.x-1,j,k) > 0)
222 ? f0(vlo.x,j,k,n) : Real(0.0);
223 Real cf1 = (j == vlo.y && m1(i,vlo.y-1,k) > 0)
224 ? f1(i,vlo.y,k,n) : Real(0.0);
225 Real cf2 = (k == vlo.z && m2(i,j,vlo.z-1) > 0)
226 ? f2(i,j,vlo.z,n) : Real(0.0);
227 Real cf3 = (i == vhi.x && m3(vhi.x+1,j,k) > 0)
228 ? f3(vhi.x,j,k,n) : Real(0.0);
229 Real cf4 = (j == vhi.y && m4(i,vhi.y+1,k) > 0)
230 ? f4(i,vhi.y,k,n) : Real(0.0);
231 Real cf5 = (k == vhi.z && m5(i,j,vhi.z+1) > 0)
232 ? f5(i,j,vhi.z,n) : Real(0.0);
233
234 Real gamma = alpha*a(i,j,k)
235 + dhx*(bX(i,j,k,n)+bX(i+1,j,k,n))
236 + dhy*(bY(i,j,k,n)+bY(i,j+1,k,n))
237 + dhz*(bZ(i,j,k,n)+bZ(i,j,k+1,n));
238
239 Real g_m_d = gamma
240 - (dhx*(bX(i,j,k,n)*cf0 + bX(i+1,j,k,n)*cf3)
241 + dhy*(bY(i,j,k,n)*cf1 + bY(i,j+1,k,n)*cf4)
242 + dhz*(bZ(i,j,k,n)*cf2 + bZ(i,j,k+1,n)*cf5));
243
244 Real rho = dhx*( bX(i ,j,k,n)*phi(i-1,j,k,n)
245 + bX(i+1,j,k,n)*phi(i+1,j,k,n) )
246 + dhy*( bY(i,j ,k,n)*phi(i,j-1,k,n)
247 + bY(i,j+1,k,n)*phi(i,j+1,k,n) )
248 + dhz*( bZ(i,j,k ,n)*phi(i,j,k-1,n)
249 + bZ(i,j,k+1,n)*phi(i,j,k+1,n) );
250
251 Real res = rhs(i,j,k,n) - (gamma*phi(i,j,k,n) - rho);
252 phi(i,j,k,n) = phi(i,j,k,n) + omega/g_m_d * res;
253 }
254}

319AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
320void tridiagonal_solve (Array1D<Real,0,31>& a_ls, Array1D<Real,0,31>& b_ls, Array1D<Real,0,31>& c_ls,
321 Array1D<Real,0,31>& r_ls, Array1D<Real,0,31>& u_ls, Array1D<Real,0,31>& gam,
322 int ilen ) noexcept
323{
324 Real bet = b_ls(0);
325 u_ls(0) = r_ls(0) / bet;
326
327 for (int i = 1; i <= ilen - 1; i++) {
328 gam(i) = c_ls(i-1) / bet;
329 bet = b_ls(i) - a_ls(i)*gam(i);
330 if (bet == 0) amrex::Abort(">>>TRIDIAG FAILED");
331 u_ls(i) = (r_ls(i)-a_ls(i)*u_ls(i-1)) / bet;
332 }
333 for (int i = ilen-2; i >= 0; i--) {
334 u_ls(i) = u_ls(i) - gam(i+1)*u_ls(i+1);
335 }
336}

338AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
339void abec_gsrb_with_line_solve (
340 Box const& box, Array4<Real> const& phi, Array4<Real const> const& rhs,
341 Real alpha, Array4<Real const> const& a,
342 Real dhx, Real dhy, Real dhz,
343 Array4<Real const> const& bX, Array4<Real const> const& bY,
344 Array4<Real const> const& bZ,
345 Array4<int const> const& m0, Array4<int const> const& m2,
346 Array4<int const> const& m4,
347 Array4<int const> const& m1, Array4<int const> const& m3,
348 Array4<int const> const& m5,
349 Array4<Real const> const& f0, Array4<Real const> const& f2,
350 Array4<Real const> const& f4,
351 Array4<Real const> const& f1, Array4<Real const> const& f3,
352 Array4<Real const> const& f5,
353 Box const& vbox, int redblack, int nc) noexcept
354{
355 const auto lo = amrex::lbound(box);
356 const auto hi = amrex::ubound(box);
357 const auto vlo = amrex::lbound(vbox);
358 const auto vhi = amrex::ubound(vbox);
359
360 // amrex::Print() << "GSRB LS " << box << " " << dhx << " " << dhy << " " << dhz << std::endl;
361
362 // idir is the direction in which we will do the tridiagonal solve --
363 // it should be the direction in which the mesh spacing is much larger
364 // than in the other directions
365 int idir = 2;
366 int ilen = hi.z - lo.z + 1;
367
368 if ( (dhx <= dhy) && (dhz <= dhy) ) {
369 idir = 1;
370 ilen = hi.y - lo.y + 1;
371 }
372 if ( (dhy <= dhx) && (dhz <= dhx) ) {
373 idir = 0;
374 ilen = hi.x - lo.x + 1;
375 }
376
377 // This assertion should be moved outside the kernel for performance!
378 if (ilen > 32) amrex::Abort("abec_gsrb_with_line_solve is hard-wired to be no longer than 32");
379
380 Array1D<Real,0,31> a_ls;
381 Array1D<Real,0,31> b_ls;
382 Array1D<Real,0,31> c_ls;
383 Array1D<Real,0,31> r_ls;
384 Array1D<Real,0,31> u_ls;
385 Array1D<Real,0,31> gam;
386
387 if (idir == 2) {
388 for (int n = 0; n < nc; ++n) {
389 for (int j = lo.y; j <= hi.y; ++j) {
390 AMREX_PRAGMA_SIMD
391 for (int i = lo.x; i <= hi.x; ++i) {
392 if ((i+j+redblack)%2 == 0) {
393
394 for (int k = lo.z; k <= hi.z; ++k)
395 {
396 Real gamma = alpha*a(i,j,k)
397 + dhx*(bX(i,j,k,n)+bX(i+1,j,k,n))
398 + dhy*(bY(i,j,k,n)+bY(i,j+1,k,n))
399 + dhz*(bZ(i,j,k,n)+bZ(i,j,k+1,n));
400
401 Real cf0 = (i == vlo.x && m0(vlo.x-1,j,k) > 0)
402 ? f0(vlo.x,j,k,n) : Real(0.0);
403 Real cf1 = (j == vlo.y && m1(i,vlo.y-1,k) > 0)
404 ? f1(i,vlo.y,k,n) : Real(0.0);
405 Real cf2 = (k == vlo.z && m2(i,j,vlo.z-1) > 0)
406 ? f2(i,j,vlo.z,n) : Real(0.0);
407 Real cf3 = (i == vhi.x && m3(vhi.x+1,j,k) > 0)
408 ? f3(vhi.x,j,k,n) : Real(0.0);
409 Real cf4 = (j == vhi.y && m4(i,vhi.y+1,k) > 0)
410 ? f4(i,vhi.y,k,n) : Real(0.0);
411 Real cf5 = (k == vhi.z && m5(i,j,vhi.z+1) > 0)
412 ? f5(i,j,vhi.z,n) : Real(0.0);
413
414 Real g_m_d = gamma
415 - (dhx*(bX(i,j,k,n)*cf0 + bX(i+1,j,k,n)*cf3)
416 + dhy*(bY(i,j,k,n)*cf1 + bY(i,j+1,k,n)*cf4)
417 + dhz*(bZ(i,j,k,n)*cf2 + bZ(i,j,k+1,n)*cf5));
418
419 Real rho = dhx*( bX(i ,j,k,n)*phi(i-1,j,k,n)
420 + bX(i+1,j,k,n)*phi(i+1,j,k,n) )
421 + dhy*( bY(i,j ,k,n)*phi(i,j-1,k,n)
422 + bY(i,j+1,k,n)*phi(i,j+1,k,n) );
423
424 // We have already accounted for this external boundary in the coefficient of phi(i,j,k,n)
425 if (i == vlo.x && m0(vlo.x-1,j,k) > 0)
426 rho -= dhx*bX(i ,j,k,n)*phi(i-1,j,k,n);
427 if (i == vhi.x && m3(vhi.x+1,j,k) > 0)
428 rho -= dhx*bX(i+1,j,k,n)*phi(i+1,j,k,n);

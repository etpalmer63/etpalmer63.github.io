31struct RedistributeUnpackPolicy
32{
33 template <class PTile>
34 void resizeTiles (std::vector<PTile*>& tiles, const std::vector<int>& sizes, std::vector<int>& offsets) const
35 {
36 int N = static_cast<int>(sizes.size());
37
38 std::map<PTile*, int> tile_sizes;
39 for(int i = 0; i < N; ++i)
40 tile_sizes[tiles[i]] = tiles[i]->numParticles();
41
42 for(int i = 0; i < N; ++i)
43 {
44 offsets.push_back(tile_sizes[tiles[i]]);
45 tile_sizes[tiles[i]] += sizes[i];
46 }
47
48 for (auto& kv : tile_sizes)
49 kv.first->resize(kv.second);
50 }

150 {
151 int gid = kv.first.first;

166 int index = Gpu::Atomic::Inc(

252template <class PC, class Buffer, std::enable_if_t<IsParticleContainer<PC>::value, int> foo = 0>
253void packBuffer (const PC& pc, const ParticleCopyOp& op, const ParticleCopyPlan& plan,
254 Buffer& snd_buffer)
255{
256 BL_PROFILE("amrex::packBuffer");
257
258 using ParticleType = typename PC::ParticleType;
259
260 Long psize = pc.superParticleSize();
261
262 int num_levels = pc.BufferMap().numLevels();
263 int num_buckets = pc.BufferMap().numBuckets();
264
265 Long total_buffer_size = 0;
266 if (plan.m_snd_offsets.size() == 0)
267 {
268 unsigned int np = 0;
269 Gpu::copy(Gpu::deviceToHost, plan.m_box_offsets.begin() + num_buckets,
270 plan.m_box_offsets.begin() + num_buckets + 1, &np);
271 total_buffer_size = np*psize;
272 }
273 else
274 {
275 total_buffer_size = plan.m_snd_offsets.back();
276 }
277 snd_buffer.resize(total_buffer_size);
278
279 auto p_comm_real = pc.d_communicate_real_comp.dataPtr();
280 auto p_comm_int = pc.d_communicate_int_comp.dataPtr();
281
282 for (int lev = 0; lev < num_levels; ++lev)
283 {
284 const auto& geom = pc.Geom(lev);
285 auto& plev = pc.GetParticles(lev);
286 const auto plo = geom.ProbLoArray();
287 const auto phi = geom.ProbHiArray();
288 const auto is_per = geom.isPeriodicArray();
289
290 for (auto& kv : plev)
291 {
292 int gid = kv.first.first;
293 int tid = kv.first.second;
294 auto index = std::make_pair(gid, tid);
295
296 auto& src_tile = plev.at(index);
297 const auto ptd = src_tile.getConstParticleTileData();
298
299 int num_copies = op.numCopies(gid, lev);
300 if (num_copies == 0) continue;
301
302 auto p_boxes = op.m_boxes[lev].at(gid).dataPtr();
303 auto p_levels = op.m_levels[lev].at(gid).dataPtr();
304 auto p_src_indices = op.m_src_indices[lev].at(gid).dataPtr();
305 auto p_periodic_shift = op.m_periodic_shift[lev].at(gid).dataPtr();
306 auto p_dst_indices = plan.m_dst_indices[lev].at(gid).dataPtr();
307 auto p_snd_buffer = snd_buffer.dataPtr();
308 GetSendBufferOffset get_offset(plan, pc.BufferMap());
309
310 AMREX_FOR_1D ( num_copies, i,
311 {
312 int dst_box = p_boxes[i];
313 if (dst_box >= 0)
314 {
315 int dst_lev = p_levels[i];
316 auto dst_offset = get_offset(dst_box, dst_lev, psize, p_dst_indices[i]);
317 int src_index = p_src_indices[i];
318 ptd.packParticleData(p_snd_buffer, src_index, dst_offset, p_comm_real, p_comm_int);
319
320 const IntVect& pshift = p_periodic_shift[i];
321 bool do_periodic_shift =
322 AMREX_D_TERM( (is_per[0] && pshift[0] != 0),
323 || (is_per[1] && pshift[1] != 0),
324 || (is_per[2] && pshift[2] != 0) );
325
326 if (do_periodic_shift)
327 {
328 ParticleType p;
329 amrex::Gpu::memcpy(&p, &p_snd_buffer[dst_offset], sizeof(ParticleType));
330 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
331 {
332 if (! is_per[idim]) continue;
333 if (pshift[idim] > 0)
334 p.pos(idim) += phi[idim] - plo[idim];
335 else if (pshift[idim] < 0)
336 p.pos(idim) -= phi[idim] - plo[idim];
337 }
338 amrex::Gpu::memcpy(&p_snd_buffer[dst_offset], &p, sizeof(ParticleType));
339 }
340 }
341 });
342 }
343 }

346template <class PC, class Buffer, class UnpackPolicy,
347 std::enable_if_t<IsParticleContainer<PC>::value, int> foo = 0>
348void unpackBuffer (PC& pc, const ParticleCopyPlan& plan, const Buffer& snd_buffer, const UnpackPolicy&& policy)
349{
350 BL_PROFILE("amrex::unpackBuffer");
351
352 using PTile = typename PC::ParticleTileType;
353
354 int num_levels = pc.BufferMap().numLevels();
355 Long psize = pc.superParticleSize();
356
357 // count how many particles we have to add to each tile
358 std::vector<int> sizes;
359 std::vector<PTile*> tiles;
360 for (int lev = 0; lev < num_levels; ++lev)
361 {
362 for(MFIter mfi = pc.MakeMFIter(lev); mfi.isValid(); ++mfi)
363 {
364 int gid = mfi.index();
365 int tid = mfi.LocalTileIndex();
366 auto& tile = pc.DefineAndReturnParticleTile(lev, gid, tid);
367 int num_copies = plan.m_box_counts_h[pc.BufferMap().gridAndLevToBucket(gid, lev)];
368 sizes.push_back(num_copies);
369 tiles.push_back(&tile);
370 }
371 }
372
373 // resize the tiles and compute offsets
374 std::vector<int> offsets;
375 policy.resizeTiles(tiles, sizes, offsets);
376
377 auto p_comm_real = pc.d_communicate_real_comp.dataPtr();
378 auto p_comm_int = pc.d_communicate_int_comp.dataPtr();
379
380 // local unpack
381 int uindex = 0;
382 for (int lev = 0; lev < num_levels; ++lev)
383 {
384 auto& plev = pc.GetParticles(lev);
385 for(MFIter mfi = pc.MakeMFIter(lev); mfi.isValid(); ++mfi)
386 {
387 int gid = mfi.index();
388 int tid = mfi.LocalTileIndex();
389 auto index = std::make_pair(gid, tid);
390
391 auto& tile = plev[index];
392
393 GetSendBufferOffset get_offset(plan, pc.BufferMap());
394 auto p_snd_buffer = snd_buffer.dataPtr();
395
396 int offset = offsets[uindex];
397 int size = sizes[uindex];
398 ++uindex;
399
400 auto ptd = tile.getParticleTileData();
401 AMREX_FOR_1D ( size, i,
402 {
403 auto src_offset = get_offset(gid, lev, psize, i);
404 int dst_index = offset + i;
405 ptd.unpackParticleData(p_snd_buffer, src_offset, dst_index, p_comm_real, p_comm_int);
406 });
407 }
408 }

411template <class PC, class Buffer,
412 std::enable_if_t<IsParticleContainer<PC>::value, int> foo = 0>
413void communicateParticlesStart (const PC& pc, ParticleCopyPlan& plan, const Buffer& snd_buffer, Buffer& rcv_buffer)
414{
415 BL_PROFILE("amrex::communicateParticlesStart");
416
417#ifdef AMREX_USE_MPI
418 Long psize = pc.superParticleSize();
419 const int NProcs = ParallelContext::NProcsSub();
420 const int MyProc = ParallelContext::MyProcSub();
421
422 if (NProcs == 1) return;
423
424 Vector<int> RcvProc;
425 Vector<Long> rOffset;
426
427 plan.m_rcv_pad_correction_h.resize(0);
428 plan.m_rcv_pad_correction_h.push_back(0);
429
430 Long TotRcvBytes = 0;
431 for (int i = 0; i < NProcs; ++i) {
432 if (plan.m_rcv_num_particles[i] > 0) {
433 RcvProc.push_back(i);
434 rOffset.push_back(TotRcvBytes);
435 Long nbytes = plan.m_rcv_num_particles[i]*psize;
436 std::size_t acd = ParallelDescriptor::alignof_comm_data(nbytes);
437 TotRcvBytes = amrex::aligned_size(acd, TotRcvBytes);
438 TotRcvBytes += amrex::aligned_size(acd, nbytes);
439 plan.m_rcv_pad_correction_h.push_back(plan.m_rcv_pad_correction_h.back() + nbytes);
440 }
441 }
442
443 for (int i = 0; i < plan.m_nrcvs; ++i)
444 {
445 plan.m_rcv_pad_correction_h[i] = rOffset[i] - plan.m_rcv_pad_correction_h[i];
446 }
447
448 plan.m_rcv_pad_correction_d.resize(plan.m_rcv_pad_correction_h.size());
449 Gpu::copy(Gpu::hostToDevice, plan.m_rcv_pad_correction_h.begin(), plan.m_rcv_pad_correction_h.end(),
450 plan.m_rcv_pad_correction_d.begin());
451
452 rcv_buffer.resize(TotRcvBytes);
453
454 plan.m_nrcvs = RcvProc.size();
455
456 plan.m_particle_stats.resize(0);
457 plan.m_particle_stats.resize(plan.m_nrcvs);
458
459 plan.m_particle_rreqs.resize(0);
460 plan.m_particle_rreqs.resize(plan.m_nrcvs);
461
462 const int SeqNum = ParallelDescriptor::SeqNum();
463
464 // Post receives.
465 for (int i = 0; i < plan.m_nrcvs; ++i) {
466 const auto Who = RcvProc[i];
467 const auto offset = rOffset[i];
468 Long nbytes = plan.m_rcv_num_particles[Who]*psize;
469 std::size_t acd = ParallelDescriptor::alignof_comm_data(nbytes);
470 const auto Cnt = amrex::aligned_size(acd, nbytes);
471
472 AMREX_ASSERT(Cnt > 0);
473 AMREX_ASSERT(Who >= 0 && Who < NProcs);
474 AMREX_ASSERT(amrex::aligned_size(acd, nbytes) % acd == 0);
475
476 plan.m_particle_rreqs[i] =
477 ParallelDescriptor::Arecv((char*) (rcv_buffer.dataPtr() + offset), Cnt, Who, SeqNum, ParallelContext::CommunicatorSub()).req();
478 }
479
480 if (plan.m_NumSnds == 0) return;
481
482 // Send.
483 for (int i = 0; i < NProcs; ++i)
484 {
485 if (i == MyProc) continue;
486 const auto Who = i;
487 const auto Cnt = plan.m_snd_counts[i];
488 if (Cnt == 0) continue;
489
490 auto snd_offset = plan.m_snd_offsets[i];
491 AMREX_ASSERT(plan.m_snd_counts[i] % ParallelDescriptor::alignof_comm_data(plan.m_snd_num_particles[i]*psize) == 0);
492 AMREX_ASSERT(Who >= 0 && Who < NProcs);
493 AMREX_ASSERT(snd_offset % ParallelDescriptor::alignof_comm_data(plan.m_snd_num_particles[i]*psize) == 0);
494
495 ParallelDescriptor::Send((char const*)(snd_buffer.dataPtr()+snd_offset), Cnt, Who, SeqNum,
496 ParallelContext::CommunicatorSub());
497 }
498#else
499 amrex::ignore_unused(pc,plan,snd_buffer,rcv_buffer);
500#endif // MPI
501}

508{
509 BL_PROFILE("amrex::unpackRemotes");
510
511#ifdef AMREX_USE_MPI

514

518
519 if (plan.m_nrcvs > 0)
520 {
521 auto p_comm_real = pc.d_communicate_real_comp.dataPtr();
522 auto p_comm_int = pc.d_communicate_int_comp.dataPtr();
523 auto p_rcv_buffer = rcv_buffer.dataPtr();

525 std::vector<int> sizes;
526 std::vector<PTile*> tiles;
527 for (int i = 0, N = plan.m_rcv_box_counts.size(); i < N; ++i)
528 {

530 int lev = plan.m_rcv_box_levs[i];
531 int gid = plan.m_rcv_box_ids[i];

536 }

539 policy.resizeTiles(tiles, sizes, offsets);
540 Gpu::Device::synchronize();
541 int uindex = 0;
542 int procindex = 0, rproc = plan.m_rcv_box_pids[0];
543 for (int i = 0, N = plan.m_rcv_box_counts.size(); i < N; ++i)

546 int gid = plan.m_rcv_box_ids[i];
547 int tid = 0;
548 auto offset = plan.m_rcv_box_offsets[i];
549 procindex = (rproc == plan.m_rcv_box_pids[i]) ? procindex : procindex+1;
550 rproc = plan.m_rcv_box_pids[i];
551

553 auto ptd = tile.getParticleTileData();

564
565 AMREX_FOR_1D ( size, ip, {
566 Long src_offset = psize*(offset + ip) + p_pad_adjust[procindex];
567 int dst_index = dst_offset + ip;
568 ptd.unpackParticleData(p_rcv_buffer, src_offset, dst_index,
569 p_comm_real, p_comm_int);
570 });

21//
22// ParmParse class implements a simple database for the storage and
23// retrieval of command-line and input-file arguments. The entries are
24// stored in a static table in (name,value_list) pairs.
25//
26// The format of the input file is a series of DEFINITIONS.
27//
28// A DEFINITION is of the form <name> = <value> <value> ...
29// The equal sign is important since the list of values can span multiple
30// lines.
31//
32// Comments in an input file include all text from a '#' character to the
33// end of the line. Here is an example input file:
34//
35// niter = 100 # niter is an integer
36// title = "Double Wammy" # example of a string with spaces
37// cell_size = 0.5 0.75 # cell spacing in each dimension
38// plot.var = Density 1 10 # a list of values
39// plot.var = Energy 5 12 # another list of values
40// bigarray = 1 2 3 4 5 6 7 8 # first part of array
41// 9 10 11 12 # continuation of bigarray
42// test = apple "boy blue" 10 20 30 40
43// FILE = prob_file # insert contents of this "prob_file" here
44//
45// The "FILE = <filename>" definition is special. Rather than just
46// adding this entry to the database, it reads the contents of <filename>
47// into the database.
48//
49// ParmParse stores all entries in a static table which is built the
50// first time a ParmParse object is constructed (usually in main()).
51// Subsequent invocations have access to this table.
52// A ParmParse constructor has an optional "prefix" argument that will
53// limit the searches to only those entries of the table with this prefix
54// in name. For example:
55// ParmParse pp("plot");
56// will find only those entries with name given by "plot.<string>".
57//
58// All values in the table are stored as strings. For example, the
59// values of "cell_size" in the above input file are stored as the
60// strings "0.5" and "0.75". These strings can be returned as either
61// strings or numeric values by the query functions.
62// Character strings with spaces must be delimited by double quotes
63// in the input file but the quotes are stripped before they are entered
64// into the table. For example, 'title' in the above input file has a
65// single value, the string 'Double Wammy' (without the quotes).
66// Each value in the list associated with a definition can be referred to
67// by its index number. The index numbers start at 0 just like an array
68// in the C programming language. Consider the definition of "test" in
69// the above input file. The first value 'apple'is a string with index
70// 0. The second value 'boy blue' is a string with index 1. The
71// remaining four values are integers indexed 2, 3, 4, and 5.
72//
73// For a string value to represent an integer or float it must fit the
74// following regular expression:
75// Sign ::= '+' | '-'
76// Digit ::= '0' | '1' | ... | '9'
77// Integer ::= [Sign]Digit+
78// Exp ::= ('e'|'E')Integer
79// Float ::= ( Integer[.Digit*][Exp] | [Integer].Digit+[Exp] )
80//
81// Where '+' indicates one or more occurrences, '*' represents zero or
82// more occurrences, '|' means one or the other and '[]' represents zero
83// or one occurrence.
84//
85// Note that floats and doubles have the same string representation and
86// that the FORTRAN "double" exponent format is not supported.
87// That is, 1.0d+3 is not a valid representation of a floating point
88// number but that 1.0e+3 is acceptable.
89//
90// There are a host of functions allowing the user to query the database
91// and retrieve values. Here are some general rules about the names of
92// the member functions:
93//
94// * Functions with the string "get" in their names attempt to get a
95// value or an array of values from the table. They generate a
96// run-time error if they are not successful.
97//
98// * Functions with the string "query" in their names attempt to get a
99// value or an array of values from the table. They return the value 1
100// (true) if they are successful and 0 (false) if not.
101//
102// * Functions with the string "kth" in their names get values from
103// the Kth entry with the given name in the database. This is
104// necessary since there may be multiple definitions with the same
105// name in the database.
106//
107// * Functions without the string "kth" in their names get values from
108// the last entry with the given name in the database. Note that the
109// definitions from the command line are appended to the database table
110// and hence will be the last entries.
111//
112// * Functions with the string "arr" in their names get an Array of
113// values from the given entry in the table. The array argument is
114// resized (if necessary) to hold all the values requested.
115//
116// * Functions without the string "arr" in their names get single
117// values from the given entry in the table.
118//
119// The following is a code sample showing how to use ParmParse:
120//
121// main(int argc, char **argv)
122// {
123// char* in_file_name = argv[1];
124// ParmParse::Initialize(argc-2, argv+2, in_file_name);
125//
126// // Query table for value of "niter". If not in table
127// // then set to default value
128// if (!pp.query("niter",niter)) niter = 20;
129//
130// // read array of cell sizes if in table
131// Vector<float> dx;
132// if (nx=pp.countval("cell_size")) {
133// // get nx values starting at index 0 and store in dx.
134// // dx is automatically resized here.
135// pp.getarr("cell_size",dx,0,nx);
136// }
137// ParmParse::Finalize();
138// }
139//
140// void do_graphics()
141// {
142// //
143// // Will only query entries with the "plot" prefix:
144// //
145// ParmParse pp("plot");
146// //
147// // Read all variables with "plot.var" keyword.
148// //
149// std::string var_name;
150// Vector<int> range;
151// int num = pp.countname("var");
152// for (int k = 0; k < num; k++)
153// {
154// //
155// // Element 0 in list is a string.
156// //
157// pp.getkth("var",k,var_name,0);
158// //
159// // Elements 1 and 2 are integers.
160// // Note that "range" will be resized to hold 2 elements.
161// //
162// pp.getktharr("var",k,range,1,2);
163// cout << "variable = " << var_name << "lo, hi = ",
164// << range[0] << " " << range[1] << endl;
165// }
166// }
167// -----------------------------------------------------------------
168// ----------------------- END COMMENTS ---------------------------
169// -----------------------------------------------------------------

916 template <typename T, std::size_t N>
917 void get (const char* name, std::array<T,N>& ref) const {
918 std::vector<T> v;
919 this->getarr(name, v);
920 AMREX_ALWAYS_ASSERT(v.size() >= N);
921 for (std::size_t i = 0; i < N; ++i) {
922 ref[i] = v[i];
923 }
924 }
925
926 template <typename T, std::size_t N>
927 int query (const char* name, std::array<T,N>& ref) const {
928 std::vector<T> v;
929 int exist = this->queryarr(name, v);
930 if (exist) {
931 AMREX_ALWAYS_ASSERT(v.size() >= N);
932 for (std::size_t i = 0; i < N; ++i) {
933 ref[i] = v[i];
934 }
935 }
936 return exist;
937 }

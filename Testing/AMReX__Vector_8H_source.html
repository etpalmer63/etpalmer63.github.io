198 template <class T, class H>
199 std::size_t removeDupDoit (Vector<T>& vec, std::size_t start, std::size_t stop)
200 {
201 std::size_t N = stop-start;
202 if (N < 2) return stop;
203
204 T* const data = vec.data() + start;
205 T const sentinel = data[0]; // duplicates will be set to sentinel and removed later
206 H const hasher;
207 for (std::size_t i = 1; i < N; ) {
208 if (data[i] == sentinel) {
209 ++i;
210 continue;
211 }
212
213 std::size_t const hash = hasher(data[i]) % N;
214 if (i == hash) { // data[i] in correct hash position
215 ++i;
216 continue;
217 }
218
219 if (data[i] == data[hash]) {
220 data[i] = sentinel; // because it's a duplicate
221 ++i;
222 continue;
223 }
224
225 if (data[hash] == sentinel) {
226 std::swap(data[hash], data[i]);
227 // after swap, new data[i] holds sentinel
228 // newdata[hash] in correct hash poitiion
229 ++i;
230 continue;
231 }
232
233 std::size_t const hashhash = hasher(data[hash]) % N;
234 if (hashhash != hash) { // data[hash] not in correct has poision, thus will yield it's position
235 std::swap(data[i], data[hash]);
236 // after swap, new data[hash] in correct hash position
237 // new data[i] not sure
238 if (hash < i) { // we have seen new data[i]
239 ++i;
240 } // else next iteration we will work on data[i]
241 } else { // data[hash] in correct hash position, but data[i] is not because of hash collision
242 ++i;
243 }
244 }
245
246 // Now there are three types for data[i]
247 // (1) sentinel
248 // (2) data[i] != sentinel and hash(data[i]) == i
249 // (3) data[i] != sentinel and hash(data[i]) != i because of hash collision
250 // All type 2s are unique, and all sentinels except one can be removed.
251 // We will move all type 2s to the beginning, all sentinels to the end.
252 // This will leave all type 3s in the middle. Then we will work on the middle
253 // part plus one sentinel.
254
255 std::size_t swapPos = 0;
256 for (std::size_t i = 0; i < N; ++i) {
257 // move type 2 to the beginning pointed to by swapPos
258 if (data[i] != sentinel && i == hasher(data[i]) % N) {
259 std::swap(data[i], data[swapPos++]);
260 }
261 }
262
263 // Now we have moved all type 2 elements to the beginning, [0,swapPos)
264
265 std::size_t sentinelPos = N;
266 for (std::size_t i = swapPos; i < sentinelPos; ) {
267 // move type 1 to the end
268 if(data[i] == sentinel) {
269 std::swap(data[i], data[--sentinelPos]);
270 } else {
271 ++i;
272 }
273 }
274
275 // recursively work on the middle part
276 return detail::removeDupDoit<T,H>(vec, start+swapPos, start+sentinelPos+1);
277 }

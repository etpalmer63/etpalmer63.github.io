219template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
220IntVect
221NeighborParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>
222::computeRefFac (const int src_lev, const int lev)
223{
224 IntVect ref_fac = IntVect(AMREX_D_DECL(1,1,1));
225 if (src_lev < lev) {
226 for (int l = src_lev; l < lev; ++l) {
227 ref_fac *= this->GetParGDB()->refRatio(l);
228 }
229 } else if (src_lev > lev) {
230 for (int l = src_lev; l > lev; --l) {
231 ref_fac *= this->GetParGDB()->refRatio(l-1);
232 }
233 ref_fac *= -1;
234 }
235 return ref_fac;
236}

238template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
239void
240NeighborParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>
241::GetCommTagsBox (Vector<NeighborCommTag>& tags, const int src_lev, const Box& in_box)
242{
243 std::vector< std::pair<int, Box> > isects;
244 Box tbx;
245
246 for (int lev = 0; lev < this->numLevels(); ++lev) {
247 Box box = in_box;
248 const IntVect& ref_fac = computeRefFac(src_lev, lev);
249 if (ref_fac < IntVect::TheZeroVector())
250 {
251 box.coarsen(-1*ref_fac);
252 }
253 else if (ref_fac > IntVect::TheZeroVector())
254 {
255 box.refine(ref_fac);
256 }
257 box.grow(computeRefFac(0, src_lev)*m_num_neighbor_cells);
258 const Periodicity& periodicity = this->Geom(lev).periodicity();
259 const std::vector<IntVect>& pshifts = periodicity.shiftIntVect();
260 const BoxArray& ba = this->ParticleBoxArray(lev);
261
262 for (auto pit=pshifts.cbegin(); pit!=pshifts.cend(); ++pit)
263 {
264 const Box& pbox = box + (*pit);
265 bool first_only = false;
266 ba.intersections(pbox, isects, first_only, 0);
267 for (const auto& isec : isects) {
268 const int grid = isec.first;
269 const int global_proc = this->ParticleDistributionMap(lev)[grid];
270 const int proc = ParallelContext::global_to_local_rank(global_proc);
271 for (IntVect iv = pbox.smallEnd(); iv <= pbox.bigEnd(); pbox.next(iv))
272 {
273 if (ba[grid].contains(iv))
274 {
275 int tile = getTileIndex(iv, ba[grid],
276 this->do_tiling, this->tile_size, tbx);
277 tags.push_back(NeighborCommTag(proc, lev, grid, tile));
278 }
279 }
280 }
281 }

287NeighborParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>
288::cacheNeighborInfo () {
289
290 BL_PROFILE("NeighborParticleContainer::cacheNeighborInfo");
291
292 AMREX_ASSERT(this->OK());
293
294 resizeContainers(this->numLevels());
295
296 clearNeighbors();
297
298 AMREX_ASSERT(hasNeighbors() == false);
299
300 const int MyProc = ParallelContext::MyProcSub();
301
302 amrex::Vector<std::map<PairIndex, Vector<NeighborIndexMap> > > local_map;
303 std::map<NeighborCommTag, Vector<NeighborIndexMap> > remote_map;
304
305 // tmp data structures used for OMP reduction
306 amrex::Vector<std::map<PairIndex, Vector<Vector<NeighborIndexMap> > > > tmp_local_map;
307 std::map<NeighborCommTag, Vector<Vector<NeighborIndexMap> > > tmp_remote_map;
308
309 local_map.resize(this->numLevels());
310 tmp_local_map.resize(this->numLevels());
311
312 int num_threads = OpenMP::get_max_threads();
313
314 for (int lev = 0; lev < this->numLevels(); ++lev) {
315 // resize our temporaries in serial
316 for (int i = 0; i < static_cast<int>(local_neighbors.size()); ++i) {
317 const NeighborCommTag& comm_tag = local_neighbors[i];
318 tmp_remote_map[comm_tag].resize(num_threads);
319 remote_map[comm_tag];
320 PairIndex index(comm_tag.grid_id, comm_tag.tile_id);
321 tmp_local_map[lev][index].resize(num_threads);
322 local_map[lev][index];
323 buffer_tag_cache[lev][index].resize(num_threads);
324 }
325 }
326
327 for (int lev = 0; lev < this->numLevels(); ++lev) {
328 // First pass - each thread collects the NeighborIndexMaps it owes to other
329 // grids / tiles / procs
330#ifdef AMREX_USE_OMP
331#pragma omp parallel
332#endif
333 {
334 Vector<NeighborCopyTag> tags;
335 tags.reserve(AMREX_D_TERM(3, *3, *3));
336 for (MyParIter pti(*this, lev); pti.isValid(); ++pti) {
337 int thread_num = OpenMP::get_thread_num();
338 const int& grid = pti.index();
339 const int& tile = pti.LocalTileIndex();
340 PairIndex src_index(grid, tile);
341
342 NeighborCopyTag src_tag(lev, grid, tile);
343
344 auto& cache = buffer_tag_cache[lev][src_index][thread_num];
345
346 auto& particles = pti.GetArrayOfStructs();
347 for (int i = 0; i < pti.numParticles(); ++i) {
348 const ParticleType& p = particles[i];
349
350 getNeighborTags(tags, p, m_num_neighbor_cells, src_tag, pti);
351
352 // Add neighbors to buffers
353 for (int j = 0; j < static_cast<int>(tags.size()); ++j) {
354 NeighborCopyTag& tag = tags[j];
355 PairIndex dst_index(tag.grid, tag.tile);
356 if (tag.grid < 0) continue;
357
358 tag.src_index = i;
359 const int cache_index = cache.size();
360 cache.push_back(tag);
361
362 const int global_who = this->ParticleDistributionMap(tag.level)[tag.grid];
363 const int who = ParallelContext::global_to_local_rank(global_who);
364 NeighborIndexMap nim(tag.level, dst_index.first, dst_index.second, -1,
365 lev, src_index.first, src_index.second,
366 cache_index, thread_num);
367 if (who == MyProc) {
368 auto& tmp = tmp_local_map[tag.level][dst_index];
369 Vector<NeighborIndexMap>& buffer = tmp[thread_num];
370 buffer.push_back(nim);
371 } else {
372 NeighborCommTag comm_tag(who, tag.level, tag.grid, tag.tile);
373 Vector<NeighborIndexMap>& buffer = tmp_remote_map[comm_tag][thread_num];
374 buffer.push_back(nim);
375 }
376 }
377 tags.clear();
378 }
379 }
380 }
381 }
382
383 for (int lev = 0; lev < this->numLevels(); ++lev) {
384 // second pass - for each tile, collect the neighbors owed from all threads
385#ifdef AMREX_USE_OMP
386#pragma omp parallel
387#endif
388 for (MFIter mfi = this->MakeMFIter(lev); mfi.isValid(); ++mfi) {
389 const int grid = mfi.index();
390 const int tile = mfi.LocalTileIndex();
391 PairIndex index(grid, tile);
392 for (int i = 0; i < num_threads; ++i) {
393 local_map[lev][index].insert(local_map[lev][index].end(),
394 tmp_local_map[lev][index][i].begin(),
395 tmp_local_map[lev][index][i].end());
396 tmp_local_map[lev][index][i].erase(tmp_local_map[lev][index][i].begin(),
397 tmp_local_map[lev][index][i].end());
398 }
399 }
400 }
401
402 // do the same for the remote neighbors
403 typename std::map<NeighborCommTag, Vector<Vector<NeighborIndexMap> > >::iterator it;
404#ifdef AMREX_USE_OMP
405#pragma omp parallel
406#pragma omp single nowait
407#endif
408 for (it=tmp_remote_map.begin(); it != tmp_remote_map.end(); it++) {
409#ifdef AMREX_USE_OMP
410#pragma omp task firstprivate(it)
411#endif
412 {
413 const NeighborCommTag& tag = it->first;
414 Vector<Vector<NeighborIndexMap> >& tmp = it->second;
415 for (int i = 0; i < num_threads; ++i) {
416 remote_map[tag].insert(remote_map[tag].end(), tmp[i].begin(), tmp[i].end());
417 tmp[i].erase(tmp[i].begin(), tmp[i].end());
418 }
419 }
420 }
421
422 for (int lev = 0; lev < this->numLevels(); ++lev) {
423 // now for the local neighbors, allocate buffers and cache
424 for (MFIter mfi = this->MakeMFIter(lev); mfi.isValid(); ++mfi) {
425 const int grid = mfi.index();
426 const int tile = mfi.LocalTileIndex();
427 PairIndex dst_index(grid, tile);
428 const Vector<NeighborIndexMap>& map = local_map[lev][dst_index];
429 const int num_ghosts = map.size();

431 this->NumRuntimeIntComps());
432 neighbors[lev][dst_index].resize(num_ghosts);
433 local_neighbor_sizes[lev][dst_index] = neighbors[lev][dst_index].size();
434 }
435 }
436
437 for (int lev = 0; lev < this->numLevels(); ++lev) {
438 for (MFIter mfi = this->MakeMFIter(lev); mfi.isValid(); ++mfi) {
439 const int grid = mfi.index();
440 const int tile = mfi.LocalTileIndex();
441 PairIndex dst_index(grid, tile);
442 const Vector<NeighborIndexMap>& map = local_map[lev][dst_index];
443 const int num_ghosts = map.size();
444#ifdef AMREX_USE_OMP
445#pragma omp parallel for
446#endif
447 for (int i = 0; i < num_ghosts; ++i) {
448 const NeighborIndexMap& nim = map[i];
449 PairIndex src_index(nim.src_grid, nim.src_tile);
450 Vector<NeighborCopyTag>& tags = buffer_tag_cache[nim.src_level][src_index][nim.thread_num];
451 AMREX_ASSERT(nim.src_index < tags.size());
452 tags[nim.src_index].dst_index = i;
453 AMREX_ASSERT(size_t(tags[nim.src_index].dst_index) < neighbors[nim.dst_level][dst_index].size());
454 }
455 }
456 }

459 std::map<int, int> tile_counts;
460 for (const auto& kv: remote_map) {
461 tile_counts[kv.first.proc_id] += 1;
462 }
463
464 for (const auto& kv: remote_map) {
465 if (kv.first.proc_id == MyProc) continue;
466 Vector<char>& buffer = send_data[kv.first.proc_id];
467 buffer.resize(sizeof(int));
468 std::memcpy(&buffer[0], &tile_counts[kv.first.proc_id], sizeof(int));
469 }
470

472 if (kv.first.proc_id == MyProc) continue;

475 Vector<char>& buffer = send_data[kv.first.proc_id];

478 buffer.resize(new_size);
479 char* dst = &buffer[old_size];

481 std::memcpy(dst, &(kv.first.grid_id ), sizeof(int)); dst += sizeof(int);
482 std::memcpy(dst, &(kv.first.tile_id ), sizeof(int)); dst += sizeof(int);
483 std::memcpy(dst, &data_size, sizeof(int)); dst += sizeof(int);

485#ifdef AMREX_USE_OMP
486#pragma omp parallel for
487#endif
488 for (int i = 0; i < np; ++i) {
489 const NeighborIndexMap& nim = kv.second[i];
490 PairIndex src_index(nim.src_grid, nim.src_tile);
491 Vector<NeighborCopyTag>& tags = buffer_tag_cache[nim.src_level][src_index][nim.thread_num];
492 tags[nim.src_index].dst_index = buffer_offset + i*cdata_size;
493 }
494 }
495
496 if ( enableInverse() )
497 {
498 for (int lev = 0; lev < this->numLevels(); ++lev)
499 {
500 for (const auto& kv : neighbors[lev])
501 {
502 inverse_tags[lev][kv.first].resize(kv.second.size());
503 }
504 }
505 }

518void
519NeighborParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>::
520getNeighborTags (Vector<NeighborCopyTag>& tags, const ParticleType& p,
521 const IntVect& nGrow, const NeighborCopyTag& src_tag, const MyParIter& pti)
522{
523 Box shrink_box = pti.tilebox();
524 shrink_box.grow(-nGrow);
525
526 if (use_mask) {
527 const BaseFab<int>& mask = (*mask_ptr[src_tag.level])[src_tag.grid];
528 AMREX_ASSERT(this->finestLevel() == 0);
529 AMREX_ASSERT(src_tag.level == 0);
530
531 const int lev = 0;
532 const IntVect& iv = this->Index(p, lev);
533 if (shrink_box.contains(iv)) return;
534
535 const Periodicity& periodicity = this->Geom(lev).periodicity();
536 const Box& domain = this->Geom(lev).Domain();
537 const IntVect& lo = domain.smallEnd();
538 const IntVect& hi = domain.bigEnd();
539
540 // Figure out all our neighbors, removing duplicates
541 AMREX_D_TERM(
542 for (int ii = -nGrow[0]; ii < nGrow[0] + 1; ii += nGrow[0]) {,
543 for (int jj = -nGrow[1]; jj < nGrow[1] + 1; jj += nGrow[1]) {,
544 for (int kk = -nGrow[2]; kk < nGrow[2] + 1; kk += nGrow[2]) {)
545 if (AMREX_D_TERM((ii == 0), && (jj == 0), && (kk == 0))) continue;
546 IntVect shift(AMREX_D_DECL(ii, jj, kk));
547 IntVect neighbor_cell = iv + shift;
548
549 NeighborCopyTag tag;
550 tag.grid = mask(neighbor_cell, MaskComps::grid);
551 tag.tile = mask(neighbor_cell, MaskComps::tile);
552 tag.level = mask(neighbor_cell, MaskComps::level);
553 if (periodicity.isAnyPeriodic()) {
554 for (int dir = 0; dir < AMREX_SPACEDIM; ++dir) {
555 if (! periodicity.isPeriodic(dir)) continue;
556 if (neighbor_cell[dir] < lo[dir])
557 tag.periodic_shift[dir] = -1;
558 else if (neighbor_cell[dir] > hi[dir])
559 tag.periodic_shift[dir] = 1;
560 }
561 }
562
563 if (tag != src_tag) tags.push_back(tag);
564
565 AMREX_D_TERM(
566 },
567 },
568 })
569
570 RemoveDuplicates(tags);
571 return;
572 }
573 else
574 {
575 std::vector< std::pair<int, Box> > isects;
576 Box tbx;
577 for (int lev = 0; lev < this->numLevels(); ++lev)
578 {
579 IntVect ref_fac = computeRefFac(0, lev);
580 const Periodicity& periodicity = this->Geom(lev).periodicity();
581 const std::vector<IntVect>& pshifts = periodicity.shiftIntVect();
582 const BoxArray& ba = this->ParticleBoxArray(lev);
583 const IntVect& iv = this->Index(p, lev);
584 for (auto pit=pshifts.cbegin(); pit!=pshifts.cend(); ++pit)
585 {
586 Box pbox = amrex::grow(Box(iv, iv), ref_fac*nGrow) + (*pit);
587 bool first_only = false;
588 ba.intersections(pbox, isects, first_only, 0);
589 for (const auto& isec : isects)
590 {
591 const Box& grid_box = ba[isec.first];
592 for (IntVect cell = pbox.smallEnd(); cell <= pbox.bigEnd(); pbox.next(cell)) {
593 if ( !grid_box.contains(cell) ) continue;
594 int tile = getTileIndex(cell, grid_box,
595 this->do_tiling, this->tile_size, tbx);
596 auto nbor = NeighborCopyTag(lev, isec.first, tile);
597 nbor.periodic_shift = -1*(*pit);
598 if (src_tag != nbor) tags.push_back(nbor);
599 }
600 }
601 }
602 }
603
604 RemoveDuplicates(tags);
605 return;
606 }

663template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
664template <class CheckPair>
665void
666NeighborParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>::
667buildNeighborList (CheckPair&& check_pair, bool /*sort*/)
668{
669 AMREX_ASSERT(numParticlesOutOfRange(*this, m_num_neighbor_cells) == 0);
670
671 resizeContainers(this->numLevels());
672
673 for (int lev = 0; lev < this->numLevels(); ++lev)
674 {
675 m_neighbor_list[lev].clear();
676
677 for (MyParIter pti(*this, lev); pti.isValid(); ++pti) {
678 PairIndex index(pti.index(), pti.LocalTileIndex());
679 m_neighbor_list[lev][index];
680 }
681
682#ifndef AMREX_USE_GPU
683 neighbor_list[lev].clear();
684 for (MyParIter pti(*this, lev); pti.isValid(); ++pti) {
685 PairIndex index(pti.index(), pti.LocalTileIndex());
686 neighbor_list[lev][index];
687 }
688#endif
689
690 auto& plev = this->GetParticles(lev);
691 const auto& geom = this->Geom(lev);
692
693#ifdef AMREX_USE_OMP
694#pragma omp parallel if (Gpu::notInLaunchRegion())
695#endif
696 for (MyParIter pti(*this, lev); pti.isValid(); ++pti)
697 {
698 int gid = pti.index();
699 int tid = pti.LocalTileIndex();
700 auto index = std::make_pair(gid, tid);
701
702 auto& ptile = plev[index];
703
704 if (ptile.numParticles() == 0) continue;
705
706 Box bx = pti.tilebox();
707 bx.grow(computeRefFac(0, lev).max()*m_num_neighbor_cells);
708
709 m_neighbor_list[lev][index].build(ptile, bx, geom,
710 std::forward<CheckPair>(check_pair),
711 computeRefFac(0, lev).max()*m_num_neighbor_cells);
712#ifndef AMREX_USE_GPU
713 const auto& counts = m_neighbor_list[lev][index].GetCounts();
714 const auto& list = m_neighbor_list[lev][index].GetList();
715
716 int li = 0;
717 for (int i = 0; i < ptile.numParticles(); ++i)
718 {
719 auto cnt = counts[i];
720 neighbor_list[lev][index].push_back(cnt);
721 for (size_t j = 0; j < cnt; ++j)
722 {
723 neighbor_list[lev][index].push_back(list[li++]+1);
724 }
725 }
726#endif
727 }
728 }
729}

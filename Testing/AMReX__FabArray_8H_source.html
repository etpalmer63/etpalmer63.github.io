59/*
60 A Collection of Fortran Array-like Objects
61
62
63 The FabArray<FAB> class implements a collection (stored as an array) of
64 Fortran array-like objects. The parameterized type FAB is intended to be
65 any class derived from BaseFab<T>. For example, FAB may be a BaseFab of
66 integers, so we could write:
67
68 FabArray<BaseFab<int> > int_fabs;
69
70 Then int_fabs is a FabArray that can hold a collection of BaseFab<int>
71 objects.
72
73 FabArray is not just a general container class for Fortran arrays. It is
74 intended to hold "grid" data for use in finite difference calculations in
75 which the data is defined on a union of (usually disjoint) rectangular
76 regions embedded in a uniform index space. This region, called the valid
77 region, is represented by a BoxArray. For the purposes of this discussion,
78 the Kth Box in the BoxArray represents the interior region of the Kth grid.
79
80 Since the intent is to be used with finite difference calculations a
81 FabArray also includes the notion of a boundary region for each grid. The
82 boundary region is specified by the ngrow parameter which tells the FabArray
83 to allocate each FAB to be ngrow cells larger in all directions than the
84 underlying Box. The larger region covered by the union of all the FABs is
85 called the region of definition. The underlying notion is that the valid
86 region contains the grid interior data and the region of definition includes
87 the interior region plus the boundary areas.
88
89 Operations are available to copy data from the valid regions into these
90 boundary areas where the two overlap. The number of components, that is,
91 the number of values that can be stored in each cell of a FAB, is either
92 given as an argument to the constructor or is inherent in the definition of
93 the underlying FAB. Each FAB in the FabArray will have the same number of
94 components.
95
96 In summary, a FabArray is an array of FABs. The Kth element contains a FAB
97 that holds the data for the Kth grid, a Box that defines the valid region
98 of the Kth grid.
99
100 A typical use for a FabArray would be to hold the solution vector or
101 right-hand-side when solving a linear system of equations on a union of
102 rectangular grids. The copy operations would be used to copy data from the
103 valid regions of neighboring grids into the boundary regions after each
104 relaxation step of the iterative method. If a multigrid method is used, a
105 FabArray could be used to hold the data at each level in the multigrid
106 hierarchy.
107
108 This class is a concrete class not a polymorphic one.
109
110 This class does NOT provide a copy constructor or assignment operator.
111*/

230void
231Copy (FabArray<FAB>& dst, FabArray<FAB> const& src, int srccomp, int dstcomp, int numcomp, const IntVect& nghost)
232{
233#ifdef AMREX_USE_GPU
234 if (Gpu::inLaunchRegion() && dst.isFusingCandidate()) {
235 auto const& srcarr = src.const_arrays();
236 auto const& dstarr = dst.arrays();
237 ParallelFor(dst, nghost, numcomp,
238 [=] AMREX_GPU_DEVICE (int box_no, int i, int j, int k, int n) noexcept
239 {
240 dstarr[box_no](i,j,k,dstcomp+n) = srcarr[box_no](i,j,k,srccomp+n);
241 });
242 Gpu::streamSynchronize();
243 } else
244#endif
245 {
246#ifdef AMREX_USE_OMP
247#pragma omp parallel if (Gpu::notInLaunchRegion())
248#endif
249 for (MFIter mfi(dst,TilingIfNotGPU()); mfi.isValid(); ++mfi)
250 {
251 const Box& bx = mfi.growntilebox(nghost);
252 if (bx.ok())
253 {
254 auto const srcFab = src.array(mfi);
255 auto dstFab = dst.array(mfi);
256 AMREX_HOST_DEVICE_PARALLEL_FOR_4D( bx, numcomp, i, j, k, n,
257 {
258 dstFab(i,j,k,dstcomp+n) = srcFab(i,j,k,srccomp+n);
259 });
260 }
261 }
262 }

1676template <class FAB>
1677void
1678FabArray<FAB>::AllocFabs (const FabFactory<FAB>& factory, Arena* ar,
1679 const Vector<std::string>& tags)
1680{
1681 const int n = indexArray.size();
1682 const int nworkers = ParallelDescriptor::TeamSize();
1683 shmem.alloc = (nworkers > 1);
1684
1685 bool alloc = !shmem.alloc;
1686
1687 FabInfo fab_info;
1688 fab_info.SetAlloc(alloc).SetShared(shmem.alloc).SetArena(ar);
1689
1690 m_fabs_v.reserve(n);
1691
1692 Long nbytes = 0L;
1693 for (int i = 0; i < n; ++i)
1694 {
1695 int K = indexArray[i];
1696 const Box& tmpbox = fabbox(K);
1697 m_fabs_v.push_back(factory.create(tmpbox, n_comp, fab_info, K));
1698 nbytes += amrex::nBytesOwned(*m_fabs_v.back());
1699 }
1700
1701 m_tags.clear();
1702 m_tags.emplace_back("All");
1703 for (auto const& t : m_region_tag) {
1704 m_tags.push_back(t);
1705 }
1706 for (auto const& t : tags) {
1707 m_tags.push_back(t);
1708 }
1709 for (auto const& t: m_tags) {
1710 updateMemUsage(t, nbytes, ar);
1711 }
1712
1713#ifdef BL_USE_TEAM
1714 if (shmem.alloc)
1715 {
1716 const int teamlead = ParallelDescriptor::MyTeamLead();
1717
1718 shmem.n_values = 0;
1719 shmem.n_points = 0;
1720 Vector<Long> offset(n,0);
1721 Vector<Long> nextoffset(nworkers,-1);
1722 for (int i = 0; i < n; ++i) {
1723 int K = indexArray[i];
1724 int owner = distributionMap[K] - teamlead;
1725 Long s = m_fabs_v[i]->size();
1726 if (ownership[i]) {
1727 shmem.n_values += s;
1728 shmem.n_points += m_fabs_v[i]->numPts();
1729 }
1730 if (nextoffset[owner] < 0) {
1731 offset[i] = 0;
1732 nextoffset[owner] = s;
1733 } else {
1734 offset[i] = nextoffset[owner];
1735 nextoffset[owner] += s;
1736 }
1737 }
1738
1739 size_t bytes = shmem.n_values*sizeof(value_type);
1740
1741 value_type *mfp;
1742 Vector<value_type*> dps;
1743
1744#if defined (BL_USE_MPI3)
1745
1746 static MPI_Info info = MPI_INFO_NULL;
1747 if (info == MPI_INFO_NULL) {
1748 MPI_Info_create(&info);
1749 MPI_Info_set(info, "alloc_shared_noncontig", "true");
1750 }
1751
1752 const MPI_Comm& team_comm = ParallelDescriptor::MyTeam().get();
1753
1754 BL_MPI_REQUIRE( MPI_Win_allocate_shared(bytes, sizeof(value_type),
1755 info, team_comm, &mfp, &shmem.win) );
1756
1757 for (int w = 0; w < nworkers; ++w) {
1758 MPI_Aint sz;
1759 int disp;
1760 value_type *dptr = 0;
1761 BL_MPI_REQUIRE( MPI_Win_shared_query(shmem.win, w, &sz, &disp, &dptr) );
1762 // BL_ASSERT(disp == sizeof(value_type));
1763 dps.push_back(dptr);
1764 }
1765
1766#else
1767
1768 amrex::Abort("BaseFab::define: to allocate shared memory, USE_MPI3 must be true");
1769
1770#endif
1771
1772 for (int i = 0; i < n; ++i) {
1773 int K = indexArray[i];
1774 int owner = distributionMap[K] - teamlead;
1775 value_type *p = dps[owner] + offset[i];
1776 m_fabs_v[i]->setPtr(p, m_fabs_v[i]->size());
1777 }
1778
1779 for (Long i = 0; i < shmem.n_values; i++, mfp++) {
1780 new (mfp) value_type;
1781 }
1782
1783 amrex::update_fab_stats(shmem.n_points, shmem.n_values, sizeof(value_type));
1784 }
1785#endif
1786}

1898 int ncomp)
1899{
1900 if (n_grow.max() > 0)
1901 {
1902#ifdef AMREX_USE_GPU
1903 if (Gpu::inLaunchRegion()) {
1904 bool use_mfparfor = true;
1905 const int nboxes = local_size();
1906 if (nboxes == 1) {
1907 if (boxarray[indexArray[0]].numPts() > Long(65*65*65)) {
1908 use_mfparfor = false;
1909 }
1910 } else {
1911 for (int i = 0; i < nboxes; ++i) {
1912 const Long npts = boxarray[indexArray[i]].numPts();
1913 if (npts >= Long(64*64*64)) {
1914 use_mfparfor = false;
1915 break;
1916 } else if (npts <= Long(17*17*17)) {
1917 break;
1918 }
1919 }
1920 }
1921 const IntVect nghost = n_grow;
1922 if (use_mfparfor) {
1923 auto const& ma = this->arrays();

1925 [=] AMREX_GPU_DEVICE (int box_no, int i, int j, int k) noexcept
1926 {
1927 auto const& a = ma[box_no];
1928 Box vbx(a);
1929 vbx.grow(-nghost);
1930 if (!vbx.contains(i,j,k)) {
1931 for (int n = 0; n < ncomp; ++n) {
1932 a(i,j,k,strt_comp+n) = val;
1933 }
1934 }
1935 });
1936 Gpu::streamSynchronize();
1937 } else {
1938 using Tag = Array4BoxTag<value_type>;
1939 Vector<Tag> tags;
1940 for (MFIter mfi(*this); mfi.isValid(); ++mfi) {
1941 Box const& vbx = mfi.validbox();
1942 auto const& a = this->array(mfi);
1943
1944 Box b;
1945#if (AMREX_SPACEDIM == 3)
1946 if (nghost[2] > 0) {
1947 b = vbx;
1948 b.setRange(2, vbx.smallEnd(2)-nghost[2], nghost[2]);
1949 b.grow(IntVect(nghost[0],nghost[1],0));
1950 tags.emplace_back(Tag{a, b});
1951 b.shift(2, vbx.length(2)+nghost[2]);
1952 tags.emplace_back(Tag{a, b});
1953 }
1954#endif
1955#if (AMREX_SPACEDIM >= 2)
1956 if (nghost[1] > 0) {
1957 b = vbx;
1958 b.setRange(1, vbx.smallEnd(1)-nghost[1], nghost[1]);

1960 tags.emplace_back(Tag{a, b});
1961 b.shift(1, vbx.length(1)+nghost[1]);
1962 tags.emplace_back(Tag{a, b});
1963 }
1964#endif
1965 if (nghost[0] > 0) {
1966 b = vbx;
1967 b.setRange(0, vbx.smallEnd(0)-nghost[0], nghost[0]);
1968 tags.emplace_back(Tag{a, b});
1969 b.shift(0, vbx.length(0)+nghost[0]);
1970 tags.emplace_back(Tag{a, b});
1971 }
1972 }
1973
1974 ParallelFor(tags, ncomp,
1975 [=] AMREX_GPU_DEVICE (int i, int j, int k, int n, Tag const& tag) noexcept
1976 {
1977 tag.dfab(i,j,k,strt_comp+n) = val;
1978 });
1979 }
1980 } else
1981#endif
1982 {
1983#ifdef AMREX_USE_OMP
1984#pragma omp parallel
1985#endif
1986 for (MFIter fai(*this); fai.isValid(); ++fai)
1987 {
1988 get(fai).template setComplement<RunOn::Host>(val, fai.validbox(), strt_comp, ncomp);
1989 }
1990 }

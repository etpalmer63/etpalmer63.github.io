268void mf_cell_quadratic_interp (int i, int j, int k, int n,
269 Array4<Real> const& fine, int fcomp,
270 Array4<Real const> const& crse, int ccomp,
271 Array4<Real const> const& slope,
272 IntVect const& ratio) noexcept

285

288 + yoff * slope(ic,jc,kc,9*n+1) // y
289 + zoff * slope(ic,jc,kc,9*n+2) // z
290 + 0.5_rt * xoff * xoff * slope(ic,jc,kc,9*n+3) // x^2
291 + 0.5_rt * yoff * yoff * slope(ic,jc,kc,9*n+4) // y^2
292 + 0.5_rt * zoff * zoff * slope(ic,jc,kc,9*n+5) // z^2
293 + xoff * yoff * slope(ic,jc,kc,9*n+6) // xy
294 + xoff * zoff * slope(ic,jc,kc,9*n+7) // xz
295 + yoff * zoff * slope(ic,jc,kc,9*n+8); // yz

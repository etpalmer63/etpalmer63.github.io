186template <class FAB, class DTOS, class Proj>
187std::enable_if_t<IsBaseFab<FAB>() && IsCallableR<Dim3, DTOS, Dim3>() && IsFabProjection<Proj, FAB>()>
188local_copy_cpu (FabArray<FAB>& dest, const FabArray<FAB>& src, int dcomp, int scomp, int ncomp,
189 FabArrayBase::CopyComTagsContainer const& local_tags, DTOS dtos, Proj proj) noexcept {
190 const int N_locs = local_tags.size();
191 if (N_locs == 0) return;
192#ifdef AMREX_USE_OMP
193#pragma omp parallel for
194#endif
195 for (int itag = 0; itag < N_locs; ++itag) {
196 const auto& tag = local_tags[itag];
197 auto const& sfab = src.const_array(tag.srcIndex);
198 auto const& dfab = dest.array (tag.dstIndex);
199 amrex::LoopConcurrentOnCpu(tag.dbox, ncomp, [=] (int i, int j, int k, int n) noexcept
200 {
201 auto const si = dtos(Dim3{i,j,k});
202 dfab(i,j,k,dcomp+n) = proj(sfab,si,scomp+n);
203 });
204 }
205}

212template <class FAB, class DTOS, class Proj>
213std::enable_if_t<IsBaseFab<FAB>() && IsCallableR<Dim3, DTOS, Dim3>() && IsFabProjection<Proj, FAB>()>
214unpack_recv_buffer_cpu (FabArray<FAB>& mf, int dcomp, int ncomp, Vector<char*> const& recv_data,
215 Vector<std::size_t> const& recv_size,
216 Vector<FabArrayBase::CopyComTagsContainer const*> const& recv_cctc,
217 DTOS dtos, Proj proj) noexcept {
218 amrex::ignore_unused(recv_size);
219
220 const int N_rcvs = recv_cctc.size();
221 if (N_rcvs == 0) return;
222
223 using T = typename FAB::value_type;
224#ifdef AMREX_USE_OMP
225#pragma omp parallel for
226#endif
227 for (int ircv = 0; ircv < N_rcvs; ++ircv) {
228 const char* dptr = recv_data[ircv];
229 auto const& cctc = *recv_cctc[ircv];
230 for (auto const& tag : cctc) {
231 auto const& dfab = mf.array(tag.dstIndex);
232 auto const& sfab = amrex::makeArray4((T const*)(dptr), tag.sbox, ncomp);
233 amrex::LoopConcurrentOnCpu(tag.dbox, ncomp, [=](int i, int j, int k, int n) noexcept {
234 auto const si = dtos(Dim3{i, j, k});
235 dfab(i, j, k, dcomp + n) = proj(sfab, si, n);
236 });
237 dptr += tag.sbox.numPts() * ncomp * sizeof(T);
238 AMREX_ASSERT(dptr <= recv_data[ircv] + recv_size[ircv]);
239 }
240 }
241}

351template <typename DTOS>
352std::enable_if_t<IsIndexMapping<DTOS>::value>
353MultiBlockCommMetaData::define (const BoxArray& dstba, const DistributionMapping& dstdm, const Box& dstbox,
354 const BoxArray& srcba, const DistributionMapping& srcdm, const IntVect& ngrow,
355 DTOS dtos) {
356 m_LocTags = std::make_unique<FabArrayBase::CopyComTagsContainer>();
357 m_SndTags = std::make_unique<FabArrayBase::MapOfCopyComTagContainers>();
358 m_RcvTags = std::make_unique<FabArrayBase::MapOfCopyComTagContainers>();
359 const int myproc = ParallelDescriptor::MyProc();
360 for (int i = 0, N = dstba.size(); i < N; ++i) {
361 const int dest_owner = dstdm[i];
362 const Box partial_dstbox = grow(dstba[i], ngrow) & dstbox;
363 if (partial_dstbox.isEmpty()) {
364 continue;
365 }
366 const Box partial_dstbox_mapped_in_src = Image(dtos, partial_dstbox).setType(srcba.ixType());
367 enum { not_first_only = 0, first_only = 1 };
368 std::vector<std::pair<int, Box>> boxes_from_src =
369 srcba.intersections(partial_dstbox_mapped_in_src, not_first_only, ngrow);
370 for (std::pair<int, Box> counted_box : boxes_from_src) {
371 const int k = counted_box.first;
372 const Box src_box = counted_box.second;
373 AMREX_ASSERT(k < srcba.size());
374 const int src_owner = srcdm[k];
375 if (dest_owner == myproc || src_owner == myproc) {
376 if (src_owner == dest_owner) {
377 const BoxList tilelist(src_box, FabArrayBase::comm_tile_size);
378 for (const Box& tilebox : tilelist) {
379 const Box inverse_image = InverseImage(dtos, tilebox).setType(dstba.ixType());
380 if ((inverse_image & partial_dstbox).ok()) {
381 m_LocTags->emplace_back(inverse_image, tilebox, i, k);
382 }
383 }
384 } else {
385 const Box inverse_image = InverseImage(dtos, src_box).setType(dstba.ixType());
386 if ((inverse_image & partial_dstbox).ok()) {
387 FabArrayBase::CopyComTagsContainer& copy_tags =
388 (src_owner == myproc) ? (*m_SndTags)[dest_owner]
389 : (*m_RcvTags)[src_owner];
390 copy_tags.emplace_back(inverse_image, src_box, i, k);
391 }
392 }
393 }
394 }
395 }

398template <class FAB, class DTOS>
399#ifdef AMREX_USE_MPI
400AMREX_NODISCARD
401#endif
402CommHandler
403Comm_nowait (FabArray<FAB>& mf, int scomp, int ncomp, FabArrayBase::CommMetaData const& cmd,
404 DTOS dtos)
405{
406#ifdef AMREX_USE_MPI
407 if (ParallelContext::NProcsSub() == 1)
408#endif
409 {
410 int N_locs = (*cmd.m_LocTags).size();
411 if (N_locs == 0) return CommHandler{};
412#ifdef AMREX_USE_GPU
413 if (Gpu::inLaunchRegion()) {
414 local_copy_gpu(mf, mf, scomp, scomp, ncomp, *cmd.m_LocTags, dtos);
415 } else
416#endif
417 {
418 local_copy_cpu(mf, mf, scomp, scomp, ncomp, *cmd.m_LocTags, dtos);
419 }
420 return CommHandler{};
421 }
422
423#ifdef AMREX_USE_MPI
424 //
425 // Do this before prematurely exiting if running in parallel.
426 // Otherwise sequence numbers will not match across MPI processes.
427 //
428 int SeqNum = ParallelDescriptor::SeqNum();
429
430 const int N_locs = cmd.m_LocTags->size();
431 const int N_rcvs = cmd.m_RcvTags->size();
432 const int N_snds = cmd.m_SndTags->size();
433
434 if (N_locs == 0 && N_rcvs == 0 && N_snds == 0) {
435 // No work to do.
436 return CommHandler{};
437 }
438
439 CommHandler handler{};
440 handler.mpi_tag = SeqNum;
441
442 if (N_rcvs > 0) {
443 handler.recv.the_data = mf.PostRcvs(*cmd.m_RcvTags, handler.recv.data, handler.recv.size,
444 handler.recv.rank, handler.recv.request, ncomp, SeqNum);
445 }
446
447 if (N_snds > 0) {
448 handler.send.the_data =
449 mf.PrepareSendBuffers(*cmd.m_SndTags, handler.send.data, handler.send.size,
450 handler.send.rank, handler.send.request, handler.send.cctc, ncomp);
451
452#ifdef AMREX_USE_GPU
453 if (Gpu::inLaunchRegion()) {
454 FabArray<FAB>::pack_send_buffer_gpu(mf, scomp, ncomp, handler.send.data,
455 handler.send.size, handler.send.cctc);
456 } else
457#endif
458 {
459 FabArray<FAB>::pack_send_buffer_cpu(mf, scomp, ncomp, handler.send.data,
460 handler.send.size, handler.send.cctc);
461 }
462
463 FabArray<FAB>::PostSnds(handler.send.data, handler.send.size, handler.send.rank, handler.send.request, SeqNum);
464 }
465
466 if (N_locs > 0)
467 {
468#ifdef AMREX_USE_GPU
469 if (Gpu::inLaunchRegion()) {
470 local_copy_gpu(mf, mf, scomp, scomp, ncomp, *cmd.m_LocTags, dtos);
471 } else
472#endif
473 {
474 local_copy_cpu(mf, mf, scomp, scomp, ncomp, *cmd.m_LocTags, dtos);
475 }
476 }
477
478 return handler;
479#endif
480}

483template <class FAB, class DTOS>
484void
485Comm_finish (FabArray<FAB>& mf, int scomp, int ncomp, FabArrayBase::CommMetaData const& cmd,
486 CommHandler handler, DTOS dtos)
487{
488 if (ParallelContext::NProcsSub() == 1) return;
489
490 const int N_rcvs = cmd.m_RcvTags->size();
491 if (N_rcvs > 0)
492 {
493 handler.recv.cctc.resize(N_rcvs, nullptr);
494 for (int k = 0; k < N_rcvs; ++k) {
495 auto const& cctc = cmd.m_RcvTags->at(handler.recv.rank[k]);
496 handler.recv.cctc[k] = &cctc;
497 }
498 handler.recv.stats.resize(handler.recv.request.size());
499 ParallelDescriptor::Waitall(handler.recv.request, handler.recv.stats);
500#ifdef AMREX_DEBUG
501 if (!CheckRcvStats(handler.recv.stats, handler.recv.size, handler.mpi_tag)) {
502 amrex::Abort("NonLocalBC::Comm_finish failed with wrong message size");
503 }
504#endif
505
506#ifdef AMREX_USE_GPU
507 if (Gpu::inLaunchRegion())
508 {
509 unpack_recv_buffer_gpu(mf, scomp, ncomp, handler.recv.data,
510 handler.recv.size, handler.recv.cctc, dtos);
511 } else
512#endif
513 {
514 unpack_recv_buffer_cpu(mf, scomp, ncomp, handler.recv.data,
515 handler.recv.size, handler.recv.cctc, dtos);
516 }
517 }
518
519 const int N_snds = cmd.m_SndTags->size();
520 if (N_snds > 0) {
521 handler.send.stats.resize(handler.send.request.size());
522 ParallelDescriptor::Waitall(handler.send.request, handler.send.stats);
523 }
524}

527template <class FAB>
528std::enable_if_t<IsBaseFab<FAB>::value>
529Rotate90 (FabArray<FAB>& mf, int scomp, int ncomp, IntVect const& nghost, Box const& domain)
530{
531 BL_PROFILE("Rotate90");
532
533 AMREX_ASSERT(domain.cellCentered());
534 AMREX_ASSERT(domain.smallEnd() == 0);
535 AMREX_ASSERT(domain.length(0) == domain.length(1));
536 AMREX_ASSERT(mf.is_cell_centered());
537 AMREX_ASSERT(scomp < mf.nComp() && scomp+ncomp <= mf.nComp());
538 AMREX_ASSERT(nghost.allLE(mf.nGrowVect()) && nghost[0] == nghost[1]);
539
540 if (nghost[0] <= 0) return;
541
542 const FabArrayBase::RB90& TheRB90 = mf.getRB90(nghost, domain);
543
544 auto handler = Comm_nowait(mf, scomp, ncomp, TheRB90,Rotate90DstToSrc{});
545
546 Box corner(-nghost, IntVect{AMREX_D_DECL(-1,-1,domain.bigEnd(2)+nghost[2])});
547#ifdef AMREX_USE_OMP
548#pragma omp parallel if (Gpu::notInLaunchRegion())
549#endif
550 for (MFIter mfi(mf); mfi.isValid(); ++mfi) {
551 Box const& bx = corner & mfi.fabbox();
552 if (bx.ok()) {
553 auto const& fab = mf.array(mfi);
554 AMREX_HOST_DEVICE_PARALLEL_FOR_4D(bx,ncomp,i,j,k,n,
555 {
556 fab(i,j,k,n) = fab(-i-1,-j-1,k,n);
557 });
558 }
559 }
560
561#ifdef AMREX_USE_MPI
562 Comm_finish(mf, scomp, ncomp, TheRB90, std::move(handler), Rotate90DstToSrc{});
563#else
564 amrex::ignore_unused(handler);
565#endif
566}

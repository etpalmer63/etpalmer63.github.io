39template <class F>
40AMREX_GPU_HOST_DEVICE
41Real
42BrentRootFinder (GpuArray<Real,AMREX_SPACEDIM> const& lo,
43 GpuArray<Real,AMREX_SPACEDIM> const& hi,
44 int rangedir, F const& f) noexcept
45{
46 const Real tol = 1.e-12;
47 const int MAXITER = 100;
48 const Real EPS = 3.0e-15;
49
50 Real p, q, r, s;
51
52 GpuArray<Real,AMREX_SPACEDIM> aPt = lo;
53 GpuArray<Real,AMREX_SPACEDIM> bPt = hi;
54
55 Real fa = IF_f(f, aPt);
56 Real fb = IF_f(f, bPt);
57 Real c = bPt[rangedir];
58 Real fc = fb;
59
60 if (fb*fa > 0.0) {
61// amrex::AllPrint() << "fa " << fa << " fb " << fb << "\n";
62 amrex::Error("BrentRootFinder. Root must be bracketed, but instead the supplied end points have the same sign.");
63 return 0.0;
64 } else if (fa == 0.0) {
65 return aPt[rangedir];
66 } else if (fb == 0.0) {
67 return bPt[rangedir];
68 }
69
70 Real d = 0.0, e = 0.0;
71 int i;
72 for (i = 0; i < MAXITER; ++i)
73 {
74 if (fb*fc > 0)
75 {
76 // Rename a, b, c and adjust bounding interval d
77 c = aPt[rangedir];
78 fc = fa;
79 d = bPt[rangedir] - aPt[rangedir];
80 e = d;
81 }
82
83 if (amrex::Math::abs(fc) < amrex::Math::abs(fb))
84 {
85 aPt[rangedir] = bPt[rangedir];
86 bPt[rangedir] = c;
87 c = aPt[rangedir];
88 fa = fb;
89 fb = fc;
90 fc = fa;
91 }
92
93 // Convergence check
94 Real tol1 = 2.0 * EPS * amrex::Math::abs(bPt[rangedir]) + 0.5 * tol;
95 Real xm = 0.5 * (c - bPt[rangedir]);
96
97 if (amrex::Math::abs(xm) <= tol1 || fb == 0.0)
98 {
99 break;
100 }
101
102 if (amrex::Math::abs(e) >= tol1 && amrex::Math::abs(fa) > amrex::Math::abs(fb))
103 {
104 // Attempt inverse quadratic interpolation
105 s = fb / fa;
106 if (aPt[rangedir] == c)
107 {
108 p = 2.0 * xm * s;
109 q = 1.0 - s;
110 }
111 else
112 {
113 q = fa / fc;
114 r = fb / fc;
115 p = s * (2.0 * xm * q * (q-r) - (bPt[rangedir]-aPt[rangedir]) * (r-1.0));
116 q = (q-1.0) * (r-1.0) * (s-1.0);
117 }
118
119 // Check whether in bounds
120 if (p > 0) q = -q;
121
122 p = amrex::Math::abs(p);
123
124 if (2.0 * p < amrex::min(3.0*xm*q-amrex::Math::abs(tol1*q), 1.0*amrex::Math::abs(e*q)))
125 {
126 // Accept interpolation
127 e = d;
128 d = p / q;
129 }
130 else
131 {
132 // Interpolation failed, use bisection
133 d = xm;
134 e = d;
135 }
136 }
137 else
138 {
139 // Bounds decreasing too slowly, use bisection
140 d = xm;
141 e = d;
142 }
143
144 // Move last best guess to a
145 aPt[rangedir] = bPt[rangedir];
146 fa = fb;
147
148 // Evaluate new trial root
149 if (amrex::Math::abs(d) > tol1)
150 {
151 bPt[rangedir] = bPt[rangedir] + d;
152 }
153 else
154 {
155 if (xm < 0) bPt[rangedir] = bPt[rangedir] - tol1;
156 else bPt[rangedir] = bPt[rangedir] + tol1;
157 }
158
159 fb = IF_f(f, bPt);
160 }
161
162 if (i >= MAXITER)
163 {
164 amrex::Error("BrentRootFinder: exceeding maximum iterations.");
165 }
166
167 return bPt[rangedir];
168}

217 int getBoxType_Cpu (const Box& bx, Geometry const& geom) const noexcept
218 {
219 const Real* problo = geom.ProbLo();
220 const Real* dx = geom.CellSize();
221 const auto& len3 = bx.length3d();
222 const int* blo = bx.loVect();
223 int nbody = 0, nzero = 0, nfluid = 0;
224 for (int k = 0; k < len3[2]; ++k) {
225 for (int j = 0; j < len3[1]; ++j) {
226 for (int i = 0; i < len3[0]; ++i) {
227 RealArray xyz {AMREX_D_DECL(problo[0]+(i+blo[0])*dx[0],
228 problo[1]+(j+blo[1])*dx[1],
229 problo[2]+(k+blo[2])*dx[2])};
230 Real v = m_f(xyz);
231 if (v == 0.0) {
232 ++nzero;
233 } else if (v > 0.0) {
234 ++nbody;
235 } else {
236 ++nfluid;
237 }
238 if (nbody > 0 && nfluid > 0) return mixedcells;
239 }
240 }
241 }
242
243 if (nbody == 0) {
244 return allregular;
245 } else if (nfluid == 0) {
246 return allcovered;
247 } else {
248 return mixedcells;
249 }

419 void updateIntercept (Array<Array4<Real>,AMREX_SPACEDIM> const& inter_arr,
420 Array<Array4<Type_t const>,AMREX_SPACEDIM> const& type_arr,
421 Array4<Real const> const& lst, Geometry const& geom) const noexcept
422 {
423 auto const& dx = geom.CellSizeArray();
424 auto const& problo = geom.ProbLoArray();
425 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
426 Array4<Real> const& inter = inter_arr[idim];
427 Array4<Type_t const> const& type = type_arr[idim];
428 const Box bx{inter};
429 AMREX_HOST_DEVICE_PARALLEL_FOR_3D (bx, i, j, k,
430 {
431 if (type(i,j,k) == Type::irregular) {
432 if (idim == 0) {
433 if (lst(i,j,k) == Real(0.0)) {
434 inter(i,j,k) = problo[0] + i*dx[0];
435 } else if (lst(i+1,j,k) == Real(0.0)) {
436 inter(i,j,k) = problo[0] + (i+1)*dx[0];
437 }
438 } else if (idim == 1) {
439 if (lst(i,j,k) == Real(0.0)) {
440 inter(i,j,k) = problo[1] + j*dx[1];
441 } else if (lst(i,j+1,k) == Real(0.0)) {
442 inter(i,j,k) = problo[1] + (j+1)*dx[1];
443 }
444 } else {
445 if (lst(i,j,k) == Real(0.0)) {
446 inter(i,j,k) = problo[2] + k*dx[2];
447 } else if (lst(i,j,k+1) == Real(0.0)) {
448 inter(i,j,k) = problo[2] + (k+1)*dx[2];
449 }
450 }
451 }
452 });
453 }

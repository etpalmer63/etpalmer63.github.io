153 template <typename N, typename F>
154 void build (BinPolicy::GPUBinPolicy, N nitems, T const* v, const Box& bx, F&& f)
155 {
156 const auto lo = lbound(bx);
157 const auto hi = ubound(bx);
158 build(BinPolicy::GPU, nitems, v, bx.numPts(),
159 [=] AMREX_GPU_DEVICE (const T& t) noexcept
160 {
161 auto iv = f(t);
162 auto iv3 = iv.dim3();
163 int nx = hi.x-lo.x+1;
164 int ny = hi.y-lo.y+1;
165 int nz = hi.z-lo.z+1;
166 index_type uix = amrex::min(nx-1,amrex::max(0,iv3.x));
167 index_type uiy = amrex::min(ny-1,amrex::max(0,iv3.y));
168 index_type uiz = amrex::min(nz-1,amrex::max(0,iv3.z));
169 return (uix * ny + uiy) * nz + uiz;
170 });
171 }

194 template <typename N, typename F>
195 void build (BinPolicy::GPUBinPolicy, N nitems, T const* v, int nbins, F&& f)
196 {
197 BL_PROFILE("DenseBins<T>::buildGPU");
198
199 m_items = v;
200
201 m_bins.resize(nitems);
202 m_perm.resize(nitems);
203
204 m_counts.resize(0);
205 m_counts.resize(nbins+1, 0);
206
207 m_offsets.resize(0);
208 m_offsets.resize(nbins+1);
209
210 index_type* pbins = m_bins.dataPtr();
211 index_type* pcount = m_counts.dataPtr();
212 amrex::ParallelFor(nitems, [=] AMREX_GPU_DEVICE (int i) noexcept
213 {
214 pbins[i] = f(v[i]);
215 Gpu::Atomic::AddNoRet(&pcount[pbins[i]], index_type{ 1 });
216 });
217
218 Gpu::exclusive_scan(m_counts.begin(), m_counts.end(), m_offsets.begin());
219
220 Gpu::copy(Gpu::deviceToDevice, m_offsets.begin(), m_offsets.end(), m_counts.begin());
221
222 index_type* pperm = m_perm.dataPtr();
223 constexpr index_type max_index = std::numeric_limits<index_type>::max();
224 amrex::ParallelFor(nitems, [=] AMREX_GPU_DEVICE (int i) noexcept
225 {
226 index_type index = Gpu::Atomic::Inc(&pcount[pbins[i]], max_index);
227 pperm[index] = i;
228 });
229
230 Gpu::Device::streamSynchronize();
231 }

298 template <typename N, typename F>
299 void build (BinPolicy::OpenMPBinPolicy, N nitems, T const* v, int nbins, F&& f)
300 {
301 BL_PROFILE("DenseBins<T>::buildOpenMP");
302
303 m_items = v;
304
305 m_bins.resize(nitems);
306 m_perm.resize(nitems);
307
308 int nchunks = OpenMP::get_max_threads();
309 int chunksize = nitems / nchunks;
310 auto counts = (index_type*)(The_Arena()->alloc(nchunks*nbins*sizeof(index_type)));
311 for (int i = 0; i < nbins*nchunks; ++i) { counts[i] = 0;}
312
313 m_counts.resize(0);
314 m_counts.resize(nbins+1, 0);
315
316 m_offsets.resize(0);
317 m_offsets.resize(nbins+1);
318
319#ifdef AMREX_USE_OMP
320#pragma omp parallel for
321#endif
322 for (int j = 0; j < nchunks; ++j) {
323 int istart = j*chunksize;
324 int istop = (j == nchunks-1) ? nitems : (j+1)*chunksize;
325 for (int i = istart; i < istop; ++i) {
326 m_bins[i] = f(v[i]);
327 ++counts[nbins*j+m_bins[i]];
328 }
329 }
330
331#ifdef AMREX_USE_OMP
332#pragma omp parallel for
333#endif
334 for (int i = 0; i < nbins; ++i) {
335 index_type total = 0;
336 for (int j = 0; j < nchunks; ++j) {
337 auto tmp = counts[nbins*j+i];
338 counts[nbins*j+i] = total;
339 total += tmp;
340 }
341 m_counts[i] = total;
342 }
343
344 // note - this part has to be serial
345 m_offsets[0] = 0;
346 for (int i = 0; i < nbins; ++i) {m_offsets[i+1] = m_offsets[i] + m_counts[i];}
347
348#ifdef AMREX_USE_OMP
349#pragma omp parallel for
350#endif
351 for (int i = 0; i < nbins; ++i) {
352 for (int j = 0; j < nchunks; ++j) {
353 counts[nbins*j+i] += m_offsets[i];
354 }
355 }
356
357#ifdef AMREX_USE_OMP
358#pragma omp parallel for
359#endif
360 for (int j = 0; j < nchunks; ++j) {
361 int istart = j*chunksize;
362 int istop = (j == nchunks-1) ? nitems : (j+1)*chunksize;
363 for (int i = istart; i < istop; ++i) {
364 auto bid = m_bins[i];
365 m_perm[counts[nbins*j+bid]++] = i;
366 }
367 }
368
369 The_Arena()->free(counts);
370 }

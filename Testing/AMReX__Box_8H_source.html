771AMREX_FORCE_INLINE
772Box&
773Box::refine (const IntVect& ref_ratio) noexcept
774{
775 if (ref_ratio != 1) {
776 IntVect shft(1);
777 shft -= btype.ixType();
778 smallend *= ref_ratio;
779 bigend += shft;
780 bigend *= ref_ratio;
781 bigend -= shft;
782 }
783 return *this;
784}

787AMREX_FORCE_INLINE
788Box&
789Box::coarsen (const IntVect& ref_ratio) noexcept
790{
791 if (ref_ratio != 1)
792 {
793 smallend.coarsen(ref_ratio);
794
795 if (btype.any())
796 {
797 IntVect off(0);
798 for (int dir = 0; dir < AMREX_SPACEDIM; dir++)
799 {
800 if (btype[dir])
801 if (bigend[dir]%ref_ratio[dir])
802 off.setVal(dir,1);
803 }
804 bigend.coarsen(ref_ratio);
805 bigend += off;
806 }
807 else
808 {
809 bigend.coarsen(ref_ratio);
810 }
811 }
812
813 return *this;
814}

1021//
1022// Modified Box is low end, returned Box is high end.
1023// If CELL: chop_pnt included in high end.
1024// If NODE: chop_pnt included in both Boxes.
1025//
1026
1027AMREX_GPU_HOST_DEVICE
1028inline
1029Box
1030Box::chop (int dir, int chop_pnt) noexcept
1031{
1032 //
1033 // Define new high end Box including chop_pnt.
1034 //
1035 IntVect sm(smallend);
1036 IntVect bg(bigend);
1037 sm.setVal(dir,chop_pnt);
1038 if (btype[dir])
1039 {
1040 //
1041 // NODE centered Box.
1042 //
1043 BL_ASSERT(chop_pnt > smallend[dir] && chop_pnt < bigend[dir]);
1044 //
1045 // Shrink original Box to just contain chop_pnt.
1046 //
1047 bigend.setVal(dir,chop_pnt);
1048 }
1049 else
1050 {
1051 //
1052 // CELL centered Box.
1053 //
1054 BL_ASSERT(chop_pnt > smallend[dir] && chop_pnt <= bigend[dir]);
1055 //
1056 // Shrink original Box to one below chop_pnt.
1057 //
1058 bigend.setVal(dir,chop_pnt-1);
1059 }
1060 return Box(sm,bg,btype);
1061}

1063AMREX_GPU_HOST_DEVICE
1064inline
1065Box&
1066Box::shiftHalf (int dir, int nzones) noexcept
1067{
1068 const int nbit = (nzones<0 ? -nzones : nzones)%2;
1069 int nshift = nzones/2;
1070 //
1071 // Toggle btyp bit if nzones is odd.
1072 //
1073 const unsigned int bit_dir = btype[dir];
1074 if (nbit)
1075 btype.flip(dir);
1076 if (nzones < 0)
1077 nshift -= (bit_dir ? nbit : 0);
1078 else
1079 nshift += (bit_dir ? 0 : nbit);
1080 smallend.shift(dir,nshift);
1081 bigend.shift(dir,nshift);
1082 return *this;
1083}

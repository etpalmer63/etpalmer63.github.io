129template <int N>
130ParserExecutor<N>
131Parser::compileHost () const
132{
133 if (m_data && m_data->m_parser) {
134 AMREX_ASSERT(N == m_data->m_nvars);
135
136 if (!(m_data->m_host_executor)) {
137 int stack_size;
138 m_data->m_exe_size = parser_exe_size(m_data->m_parser, m_data->m_max_stack_size,
139 stack_size);
140
141 if (m_data->m_max_stack_size > AMREX_PARSER_STACK_SIZE) {
142 amrex::Abort("amrex::Parser: AMREX_PARSER_STACK_SIZE, "
143 + std::to_string(AMREX_PARSER_STACK_SIZE) + ", is too small for "
144 + m_data->m_expression);
145 }
146 if (stack_size != 0) {
147 amrex::Abort("amrex::Parser: something went wrong with parser stack! "
148 + std::to_string(stack_size));
149 }
150
151 m_data->m_host_executor = (char*)The_Pinned_Arena()->alloc(m_data->m_exe_size);
152
153 try {
154 parser_compile(m_data->m_parser, m_data->m_host_executor);
155 } catch (const std::runtime_error& e) {
156 throw std::runtime_error(std::string(e.what()) + " in Parser expression \""
157 + m_data->m_expression + "\"");
158 }
159 }
160
161#ifdef AMREX_USE_GPU
162 return ParserExecutor<N>{m_data->m_host_executor, m_data->m_device_executor};
163#else
164 return ParserExecutor<N>{m_data->m_host_executor};
165#endif
166 } else {
167 return ParserExecutor<N>{};
168 }
169}

171template <int N>
172ParserExecutor<N>
173Parser::compile () const
174{
175 auto exe = compileHost<N>();
176
177#ifdef AMREX_USE_GPU
178 if (m_data && m_data->m_parser && !(m_data->m_device_executor)) {
179 m_data->m_device_executor = (char*)The_Arena()->alloc(m_data->m_exe_size);
180 Gpu::htod_memcpy_async(m_data->m_device_executor, m_data->m_host_executor,
181 m_data->m_exe_size);
182 Gpu::streamSynchronize();
183 exe.m_device_executor = m_data->m_device_executor;
184 }
185#endif
186
187 return exe;
188}
